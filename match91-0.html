def row_rotation(exam_week, seat_status):
  row, col = len(seat_status), len(seat_status[0])
  num_rotation = exam_week % row
  for i in range(row):
    for j in range(col):
      if seat_status[i][j] == "AA":
        row_aa = i
        break
  new_row = (row_aa - num_rotation) % row
  return new_row

def compress_matrix(mat):
  row, col = len(mat), len(mat[0])
  new = np.zeros((row // 2, col // 2), dtype = int)


  for i in range(0, row, 2):
    for j in range(0, col, 2):
      new[i // 2][j // 2] = mat[i][j] + mat[i][j + 1]+ mat[i + 1][j] + mat[i + 1][j + 1]
  return new

def play_game(arena):


  row, col = len(arena), len(arena[0])
  total = 0
  for i in range(row):
    for j in range(col):
      if arena[i][j] != 0 and arena[i][j] % 50 == 0:
        if i > 0 and arena[i - 1][j] == 2:
          total += 2
        if i < row - 1 and arena[i + 1][j] == 2:
          total += 2
        if j > 0 and arena[i][j - 1] == 2:
          total += 2
        if j < col -1 and arena[i][j + 1] == 2:
          total += 2

  if total >= 10:
    print(f"Points gained: {total}. Your team has survived the game.")
  else:
    print(f"points gained: {total}. Your team is out.")
