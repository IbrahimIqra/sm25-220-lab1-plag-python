def row_rotation(exam_week, seat_status):
  r,c=seat_status.shape
  for i in range(1,exam_week):
    n=np.empty(c,'object')
    for i in range(c):
      n[i]=seat_status[r-1][i]
    for i in range(r-1,0,-1):
      for j in range(c):
        seat_status[i][j]=seat_status[i-1][j]
    for i in range(len(n)):
      seat_status[0][i]=n[i]

def compress_matrix(mat):
  
  r,c=mat.shape
  res_r=r//2
  res_c=c//2
  res_mat=np.zeros((res_r,res_c),dtype=int)
  for i in range(0,r,2):
    for j in range(0,c,2):
      res_mat[i//2][j//2]=mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
  return res_mat
