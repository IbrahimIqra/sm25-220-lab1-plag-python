def compress_matrix(mat):
  r_s = len(mat)
  c_s = len(mat[0])
  new_r_s = r_s//2
  new_c_s = c_s//2

  result = np.zeros((new_r_s,new_c_s),dtype=int)


  for i in range(0,r_s,2):
    for j in range(0,c_s,2):
      s_block = (mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1])

      result[i//2][j//2] = s_block

  return result
