def mostWater(arr):
    left = 0
    right = len(arr) - 1
    maxarea = 0
    while left < right:
        min_height = min(arr[left], arr[right])
        area = min_height * (right - left)


        maxarea = max(maxarea, area)
        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1
    print(maxarea)

def compress_matrix(mat):
  row,col=mat.shape
  out=np.zeros((row//2,col//2),dtype=int)
  outr=0
  for r in range(0,row,2):
    outc=0


    for c in range(0,col,2):
      sum=mat[r][c]+mat[r][c+1]+mat[r+1][c]+mat[r+1][c+1]
      out[outr][outc]=sum
      outc+=1
    outr+=1
  return out

def play_game(arena):
    rows, cols = arena.shape
    total_points = 0
    for i in range(rows):


        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:
                points = 0
                if i > 0 and arena[i-1][j] == 2:
                    points += 1
                if i < rows - 1 and arena[i+1][j] == 2:
                    points += 1
                if j > 0 and arena[i][j-1] == 2:
                    points += 1
                if j < cols - 1 and arena[i][j+1] == 2:
                    points += 1
                total_points += points * 2
    print(f"Points Gained: {total_points}.", end=" ")
    if total_points>= 10:
        print("Your team has survived the game.")
    else:
        print("Your team is out.")
