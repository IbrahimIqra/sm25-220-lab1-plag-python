def mostWater(arr):
    l=0
    r=len(arr)-1
    max=0
    while l<r:
        if arr[l]<arr[r]:
            hei=arr[l]
        else:
            hei=arr[r]
        wid=r-l
        area=hei*wid
        if area>max:
            max=area
        elif arr[l]<arr[r]:
            l+=1
        else:
            r-=1
    print(max)

def row_rotation(exam_week, seat_status):
    ro=seat_status.shape[0]
    co=seat_status.shape[1]
    idx=0
    for i in range(ro):
        if "AA" in seat_status[i]:
            idx=i
            break


    rot=(exam_week-1)%ro
    for i in range(rot):
        lst_ro=seat_status[-1]
        for k in range(ro-1,0,-1):
            seat_status[k]=seat_status[k-1]
        seat_status[0]=lst_ro 
    newidx=(idx+rot)%ro
    print_matrix(seat_status)
    return newidx+1

def play_game(arena):
    ro=len(arena)
    co=len(arena[0])
    totpoint=0
    for i in range(ro):
      for j in range(co):
          if arena[i][j]%50==0 and arena[i][j]!=0: 
              if i>0 and arena[i-1][j]==2:
                    totpoint+=2


              if i<ro-1 and arena[i+1][j]==2:
                    totpoint+=2
              if j>0 and arena[i][j-1]==2:
                    totpoint+=2
              if j<co-1 and arena[i][j+1]==2:
                    totpoint+=2
    if totpoint>=10:
        print(f"Points Gained: {totpoint}.your team has survived the game.")
    else:
        print(f"Points Gained: {totpoint}.your team is out.")
