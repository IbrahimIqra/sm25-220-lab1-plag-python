Assignment Task 1: Container with Most Water

def mostWater(arr):
  #To Do
  area = 0
  left_count = 0
  right_count = len(arr)-1
  while left_count < right_count:
    if arr[left_count] < arr[right_count]:
        height = arr[left_count]
    else:
        height = arr[right_count]
    width = right_count - left_count

    temp = height * width



    if temp > area:
        area = temp
    if arr[left_count] < arr[right_count]:
        left_count += 1
    else:
        right_count -= 1

    print(area)

Assignment Task 3: Matrix Compressions

def compress_matrix(mat):
  #TO DO
  row = len(mat)
  com_row = row // 2
  col = len(mat[0])


  com_col = col // 2
  new_arr = np.zeros((com_row, com_col), dtype=int)
  for i in range (0,row,2):
    for j in range(0,col,2):
        sum = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]


        new_arr[i//2][j//2] = sum
  return new_arr

def play_game(arena):
    row=len(arena)
    col=len(arena[0])
    sum=0
    for i in range(row):
        for j in range(col):


            if arena[i][j]%50 == 0:
                if i+1<row and arena[i+1][j] == 2:
                    sum += 2
                if i-1<-1 and arena[i-1][j] == 2:
                    sum += 2
                if j+1<col and arena[i][j+1] == 2:
                    sum += 2
                if j-1>col and arena[i][j-1] == 2:
                    sum += 2
    if sum < 10:
        print(f'points gained: {sum}. Your team is out.')
    else:
        print(f'points gained: {sum}. Your team has survived the game.')
