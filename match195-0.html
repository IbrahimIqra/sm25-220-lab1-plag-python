def compress_matrix(mat):
    rows,cols = mat.shape
    new_row = rows // 2
    new_col = cols // 2
    result = np.zeros((new_row, new_col), dtype=int)


    for i in range(0, rows, 2):
        for j in range(0, cols, 2):


            sum = (mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1])
            result[i//2][j//2] = sum
    return result

def play_game(arena):
    rows, cols = arena.shape
    points = 0


    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:


                if i > 0 and arena[i-1][j] == 2:
                    points += 2
                if i < rows-1 and arena[i+1][j] == 2:
                    points += 2
                if j > 0 and arena[i][j-1] == 2:
                    points += 2
                if j < cols-1 and arena[i][j+1] == 2:
                    points += 2
    if points >= 10:
        print("Points Gained: {}. Your team has survived the game.".format(points))
    else:
        print("Points Gained: {}. Your team is out.".format(points))
