import numpy as np


def mostWater(arr):
    outer_max_areas = []


    for i in range(len(arr)):
        inner_areas = []


        for j in range(len(arr)):
            if i == j:
                continue
            height = min(arr[i], arr[j])
            width = abs(j - i)
            area = height * width
            inner_areas.append(area)


        max_inner = max(inner_areas)
        outer_max_areas.append(max_inner)


    print(max(outer_max_areas))

import numpy as np




def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0])


    compressed_rows = rows // 2
    compressed_cols = cols // 2


    compressed = np.zeros((compressed_rows, compressed_cols), dtype=int)


    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            block_sum = (
                mat[i][j] +
                mat[i][j+1] +
                mat[i+1][j] +
                mat[i+1][j+1]
            )
            compressed[i//2][j//2] = block_sum


    return compressed

import numpy as np




def play_game(arena):
    rows = len(arena)
    cols = len(arena[0])
    points = 0


    for i in range(rows):
        for j in range(cols):
            if arena[i][j] not in [0, 1, 2]:


                if i > 0 and arena[i-1][j] == 2:
                    points += 2
                if i < rows - 1 and arena[i+1][j] == 2:
                    points += 2
                if j > 0 and arena[i][j-1] == 2:
                    points += 2
                if j < cols - 1 and arena[i][j+1] == 2:
                    points += 2


    if points >= 10:
        print(f'Points Gained: {points}. Your team has survived the game.')
    else:
        print(f'Points Gained: {points}. Your team is out.')
