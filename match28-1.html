Task - 2 : Row Rotation Policy of Your Classroom 


def row_rotation(exam_week, seat_status):
    for _ in range(exam_week):
        last_row = seat_status[-1]


        for i in range(len(seat_status) - 1, 0, -1):
            seat_status[i] = seat_status[i - 1]
        seat_status[0] = last_row
    
    for row in range(len(seat_status)):
        if 'AA' in seat_status[row]:
            return row + 1






Task - 3 : Matrix Compression 

def compress_matrix(mat):
    rows = len(mat) // 2
    cols = len(mat[0]) // 2
    compressed = [[0 for _ in range(cols)] for _ in range(rows)]
    
    for i in range(0, len(mat), 2):
        for j in range(0, len(mat[0]), 2):


            compressed[i//2][j//2] = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
    
    return compressed





Task - 4 : Game Arena 



def play_game(arena):
    total_points = 0
    rows = len(arena)
    cols = len(arena[0]) if rows > 0 else 0

    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:
                if i > 0 and arena[i-1][j] == 2:
                    total_points += 2
                if i < rows - 1 and arena[i+1][j] == 2:
                    total_points += 2
                if j > 0 and arena[i][j-1] == 2:
                    total_points += 2
                if j < cols - 1 and arena[i][j+1] == 2:
                    total_points += 2

    if total_points < 10:
        print(f"Points Gained: {total_points}. Your team is out.")
    else:
        print(f"Points Gained: {total_points}. Your team has survived the game.")
