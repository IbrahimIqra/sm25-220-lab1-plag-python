def mostWater(height):
    lft = 0
    rgt = len(height) - 1
    max_area = 0


    while lft < rgt:
       
        h = min(height[lft], height[rgt])
        w = rgt - lft


        area = h * w
        max_area = max(max_area, area)


        if height[lft] < height[rgt]:
            lft += 1
        else:
            rgt -= 1


    print(max_area)

def compress_matrix(mat):
    row_no = len(mat)
    col_no = len(mat[0]) if row_no > 0 else 0
    compressed = []
   
   
    for i in range(0, row_no, 2):
        compressed_row = []


        for k in range(0, col_no, 2):
           
            total = mat[i][k] + mat[i][k+1] + mat[i+1][k] + mat[i+1][k+1]
            compressed_row.append(total)
        compressed.append(compressed_row)
   
    return compressed

def play_game(arena):
    totl_pts = 0
    row_no = len(arena)
    if row_no == 0:
        print("Points Gained: 0. Your team is out.")
        return
   
    col_no = len(arena[0])
   


    for i in range(row_no):
        for j in range(col_no):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:  
                pts = 0
               
                if i > 0 and arena[i-1][j] == 2:
                    pts += 1
               
                if i < row_no - 1 and arena[i+1][j] == 2:
                    pts += 1
               
                if j > 0 and arena[i][j-1] == 2:
                    pts += 1
               
                if j < col_no - 1 and arena[i][j+1] == 2:
                    pts += 1
                totl_pts += pts * 2  
   
    if totl_pts >= 10:
        print(f"Points Gained: {totl_pts}. Your team has survived the game.")
    else:
        print(f"Points Gained: {totl_pts}. Your team is out.")
