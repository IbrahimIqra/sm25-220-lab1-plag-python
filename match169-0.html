def mostWater(arr):
  left = 0
  right = len(arr) - 1
  max_area = 0
  while left < right:
    width = right - left


    if arr[left] < arr[right]:
      height = arr[left]
    else:
      height = arr[right]
    area = height * width
    if area > max_area:
      max_area = area


    if area > max_area:
      max_area = area
    if arr[left] < arr[right]:
      left += 1
    else:
      right -= 1

  rows = len(seat_status)
  cols = len(seat_status[0])
  rotated = np.array([[""] * cols for _ in range(rows)])
  for i in range(rows):
    new_index = (i + exam_week) % rows
    for j in range(cols):
        rotated[new_index][j] = seat_status[i][j]
  for i in range(rows):
    for j in range(cols):
        seat_status[i][j] = rotated[i][j]

    for j in range(cols):
      if seat_status[i][j] == "AA":

import numpy as np


def compress_matrix(mat):
  rows = len(mat)
  cols = len(mat[0])


  c_rows = rows // 2
  c_cols = cols // 2


  compressed = np.zeros((c_rows,c_cols), dtype=int)


  i = 0
  while i < rows:
    j = 0
    while j < cols:
      sum1 = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
      compressed[i//2][j//2] =sum1
      j += 2
    i += 2


  return compressed

import numpy as np


def play_game(arena):


    rows = len(arena)
    cols = len(arena[0])
    score = 0
    i = 0
    while i < rows:
      j = 0
      while j < cols:
          val = arena[i][j]
          if val % 50 == 0 and val != 0:
              count = 0
              if i - 1 >= 0 and arena[i - 1][j] == 2:
                  count += 1
              if i + 1 < rows and arena[i + 1][j] == 2:
                  count += 1
              if j - 1 >= 0 and arena[i][j - 1] == 2:
                  count += 1
              if j + 1 < cols and arena[i][j + 1] == 2:
                  count += 1
              score += count * 2
          j += 1
      i += 1


    print("Points Gained:", score, end=". ")
    if score >= 10:
        print("Your team has survived the game.")
    else:
        print("Your team is out.")
