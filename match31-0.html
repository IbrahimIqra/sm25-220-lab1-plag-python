import numpy as np




def mostWater(arr):
    left = 0
    right = len(arr) - 1
    max_water = 0


    while left < right:
       
        width = right - left
        current_arr = min(arr[left], arr[right])
        current_area = width * current_arr
        max_water = max(max_water, current_area)


       
        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1


    return max_water

def row_rotation(exam_week, seat_status):
    rows = len(seat_status)




    rotation = (exam_week - 1) % rows
    rotated = list(seat_status[-rotation:]) + list(seat_status[:-rotation])




    for row in rotated:
        print("| " + " | ".join(f"{cell:<2}" for cell in row) + " |")




    for idx, row in enumerate(rotated):
        if "AA" in row:
            print(f"\nYour friend AA will be on row {idx + 1}")

import numpy as np


def compress_matrix(mat):
    rows = len(mat)


    cols = len(mat[0])
    compressed = np.zeros((rows // 2, cols // 2), dtype=int)


    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            total = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]


            compressed[i//2][j//2] = total


    return compressed

def play_game(arena):
    rows = len(arena)


    cols = len(arena[0])
    total_points = 0


    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:  # Player position
                points = 0
                # Check above
                if i > 0 and arena[i - 1][j] == 2:
                    points += 2
                # Check below
                if i < rows - 1 and arena[i + 1][j] == 2:
                    points += 2
                # Check left
                if j > 0 and arena[i][j - 1] == 2:
                    points += 2
                # Check right
                if j < cols - 1 and arena[i][j + 1] == 2:
                    points += 2
                total_points += points


    print(f"Points Gained: {total_points}.", end=" ")
    if total_points >= 10:
        print("Your team has survived the game.")
    else:
        print("Your team is out.")
