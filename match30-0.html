def mostWater(arr):

    leftSide = 0  
    rightSide = len(arr) - 1  
    max_water = 0  

    while leftSide < rightSide:      
        current_water = min(arr[leftSide], arr[rightSide]) * (rightSide - leftSide)
      
        if current_water > max_water:
            max_water = current_water
       
        elif arr[leftSide] < arr[rightSide]:
            leftSide += 1
        else:
            rightSide -= 1   
    print(max_water)

  col=mat.shape[1]
  temp=np.zeros((row//2,col//2),dtype=int) 
                                         
  for i in range(0,row,2): 
    for j in range(0,col,2):
      sum=mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
      temp[i//2][j//2]=sum

def play_game(arena):
  row,col=arena.shape
  points=0
  for r in range(0,row):
    for c in range(0,col):
     if arena[r][c]%50==0:
      for i in range(r-1, r+2):
        if i==2:
          points+=2
  if points > 10:
        print(f'Points Gained: {points}. Your team has survived the game.')
  else:
        print(f'Points Gained: {points}. Your team is out.')
