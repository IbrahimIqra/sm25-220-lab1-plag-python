def mostWater(arr):
   
    left = 0
    right = len(arr) - 1


    max_area = 0


    while left < right:
        height = min(arr[left], arr[right])    
        width = right - left
       
        current_area = height * width
       
        max_area = max(max_area, current_area)


     
        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1


    print(max_area)

def row_rotation(exam_week, seat_status):
    rotated_matrix = np.roll(seat_status, exam_week, axis=0)


    rotated_matrix[rotated_matrix == "AA"] = "--"


    for i in range(rotated_matrix.shape[0]):
        for j in range(rotated_matrix.shape[1]):
            if rotated_matrix[i][j] == 'AA':
                return i  


    return None

def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0])
    result = np.zeros((rows//2, cols//2), dtype=int)
   
    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            total = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            result[i//2][j//2] = total
    return result


def print_matrix(mat):
    for row in mat:
        line = ""
        for val in row:
            line += "|{:4d} ".format(val)
        line += "|"
        print(line)
        print("-" * len(line))

def play_game(arena):
    total_points = 0
    rows, cols = arena.shape
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]




    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:
                points = 0
                for di, dj in directions:


                    ni, nj = i + di, j + dj
                    if 0 <= ni < rows and 0 <= nj < cols and arena[ni][nj] == 2:
                        points += 2
                total_points += points


    if total_points >= 10:
        return f'Points Gained: {total_points}. Your team has survived the game.'
    else:
        return f'Points Gained: {total_points}. Your team is out.'
