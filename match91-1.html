def mostWater(arr):
    max_water=0
    for i in range(len(arr)-1):
        for j in range(i+1,len(arr)):


            if arr[i]<arr[j]:
                cur_height=arr[i]
            else:
                cur_height=arr[j]
            cur_width=j-i
            cur_water=cur_height*cur_width
            if cur_water>max_water:
                max_water=cur_water
    print(max_water)

def compress_matrix(mat):
    row=len(mat)
    col=len(mat[0])
    result = [[0 for _ in range(col//2)] for _ in range(row//2)]


    for i in range(0, row, 2):
        for j in range(0, col, 2):
            result[i//2][j//2] = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
    return result

def play_game(arena):
    rows=len(arena)


    cols=len(arena[0])
    points=0
    for row in range(rows):
        for col in range(cols):
            if arena[row][col]>0 and arena[row][col]%50==0:             
                if row>0 and arena[row-1][col]==2:
                    points+=2
                if row<rows-1 and arena[row+1][col]==2:
                    points += 2
                if col>0 and arena[row][col- 1]==2:
                    points+=2
                if col<cols-1 and arena[row][col+1]==2:
                    points += 2
        if points>=10:
            print("points Gained:",points,"Your team has survived the game.")
        else:
            print("points Gained:",points,"Your team out.")
