def compress_matrix(mat):
  #TO DO
    m = len(mat)
    n = len(mat[0])
    row = m // 2
    col = n // 2
    result = np.zeros((row, col), dtype=int)


    for i in range(row):
        for j in range(col):
            n_block = (
                mat[2 * i][2 * j] + mat[2 * i][2 * j + 1] +
                mat[2 * i + 1][2 * j] + mat[2 * i + 1][2 * j + 1]
            )
            result[i][j] = n_block
    print('Matrix Compression:')
    return result
