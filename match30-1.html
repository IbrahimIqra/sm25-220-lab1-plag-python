def mostWater(arr):
  water=0
  x=0
  y=len(arr)-1
  while x<y:
    new_water=min(arr[x],arr[y])*(y-x)
    water=max(water,new_water)
    if arr[x]<arr[y]:
      x+=1
    else:
      y-=1
  print(water)

def compressed_matrix(mat):
  r=mat.shape[0]
  c=mat.shape[1]
  new=np.zeros((r//2,c//2),dtype=int)
  for j in range(0,r,2):
    for i in range(0,c,2):
      tot=mat[j][i]+mat[j][i+1]+mat[j+1][i]+mat[j+1][i+1]
      new[j//2][i//2]=tot
      tot=0
  return new

def play_game(arena):
  r,c=arena.shape
  p=0
  for x in range(0,r):
    for y in range(0,c):
      if arena[x][y]%50==0:
        for z in range(x-1,x+2):
          if z==2:
            p+=2
  if p>10:
    print(f'Points Gained: {p}. Your team has survived the game.')
  else:
    print(f'Points Gained: {p}. Your team is out.')
