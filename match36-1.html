def row_rotation(exam_week, seat_status):
    total_rows, total_cols = seat_status.shape




    for rotation in range(exam_week - 1):
        placeholder_row = np.array(["  "] * total_rows)


        for col in range(total_cols):
            placeholder_row[col] =seat_status[total_rows - 1][col]


        for row in range(total_rows - 1, 0, -1):
            for col in range(total_cols):
                seat_status[row][col] =seat_status[row - 1][col]


        for col in range(total_cols):
            seat_status[0][col] = placeholder_row[col]


    for row in range(total_rows):
        if seat_status[row][0] == 'Z':

def compress_matrix(mat):
  row, col = mat.shape
  result = np.zeros((row // 2, col // 2), dtype=int)
  row_pos = 0
  for i in range(0, row, 2):
    col_pos = 0
    for j in range(0, col, 2):


      result[row_pos][col_pos] = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
      col_pos += 1
    row_pos += 1


  return result
