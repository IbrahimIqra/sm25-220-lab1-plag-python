def mostWater(arr):
    n = len(arr)
    max = 0

    for i in range(n):
        for j in range(i + 1, n):


            if arr[i] < arr[j]:
                height = arr[i]
            else:
                height = arr[j]
            width = j - i
            area = height * width

            if area > max:
                max = area

    print(max)

def compress_matrix(mat):
  row, col = mat.shape
  new_row = row // 2
  new_col = col // 2
  result = np.zeros((new_row, new_col), dtype=int)
  x = 0
  for i in range(0, row, 2):
    y = 0


    for j in range(0, col, 2):
      total = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
      result[x][y] = total
      y += 1
    x += 1
  return result

def play_game(arena):
  row, col = arena.shape


  count = 0
  for i in range(row):
    for j in range(col):
      if arena[i][j] % 50 == 0 and arena[i][j] > 0:
        points = 0
        if i > 0:
          if arena[i-1][j] == 2:
            points += 2
        if i < row-1:
          if arena[i+1][j] == 2:
            points += 2
        if j > 0:
          if arena[i][j-1] == 2:
            points += 2
        if j < col-1:
          if arena[i][j+1] == 2:
            points += 2
        count += points
  if count >= 10:
    print(f'Points Gained: {count}. Your team has survived the game.')
  else:
    print(f'Points Gained: {count}. Your team is out.')
