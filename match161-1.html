def mostWater(arr):
  left=0
  right=len(arr)-1
  most_water=0
  
  while left<right:
    if arr[left]<arr[right]:
      height=arr[left]
      width=right-left
      left+=1
    else:
      height=arr[right]
      width=right-left
      right-=1
    if most_water<width*height:
      most_water=width*height
  print(most_water)

def compress_matrix(mat):
  row,colum=mat.shape
  arr=np.zeros((row//2,colum//2),dtype=int)
  sum=0


  for i in range(0,row,2):
    for j in range(0,colum,2):
      sum=mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
      arr[i//2][j//2]=sum
  return arr
