def mostWater(arr):
    left = 0
    right = len(arr) - 1
    max_water = 0

    while left < right:
        height = min(arr[left], arr[right])
        width = right - left
        max_water = max(max_water, height * width)

        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1

    print(max_water)

def compress_matrix(mat):
    row, col = len(mat), len(mat[0])
    n_row, n_col = row // 2, col // 2


    com_mat = np.zeros((n_row, n_col), dtype=int)

    for i in range(n_row):
        for j in range(n_col):
            com_mat[i][j] = mat[2 * i][2 * j] + mat[2 * i][2 * j + 1] + mat[2 * i + 1][2 * j] + mat[2 * i + 1][2 * j + 1]

    return com_mat
