def mostWater(arr):
    left = 0
    right = len(arr)-1
    final_area = 0
    current_height = 0



    while left < right:
        width = right - left
        if arr[left] < arr[right]:
            current_height = arr[left]

        else:
            current_height = arr[right]

        present_area = current_height * width

        if present_area > final_area:
            final_area = present_area

        if arr[left] < arr[right]:

            left += 1
        else:
            right -= 1

    print(final_area)

def compress_matrix(mat):

    x , y = 0, 0
    for i in range(0,row,2):
        for j in range(0,col,2):


            sum += mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]
            new_mat[x][y] = sum
            y += 1
            sum = 0
        y = 0
        x += 1
    return new_mat
