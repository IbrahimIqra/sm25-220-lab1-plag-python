def maxArea(height):
    left_line = 0
    right_line = len(height) - 1
    max_area = 0


    while left_line < right_line:
        cur_height = min(height[left_line], height[right_line])
        cur_width = right_line - left_line
        cur_area = cur_height * cur_width


        if cur_area > max_area:
            max_area = cur_area
        if height[left_line] < height[right_line]:
            left_line += 1
        else:
            right_line -= 1


    return max_area

def row_rotation(exam_week, seat_status):
    seats = [row.copy() for row in seat_status]
    for _ in range(exam_week):
        last_row = seats[-1]
        for i in range(len(seats)-1, 0, -1):
            seats[i] = seats[i-1]
        seats[0] = last_row
   
    for row_num, row in enumerate(seats, 1):
        if "AA" in row:
            return row_num
    return -1
