def play_game(arena):
  import numpy as np
 toal_points = 0
 row, col = len(arena), len(arena[0])


 for i in range(row):
     for j in range(col):
         if arena[i][j] % 50 == 0 and arena[i][j] != 0:
             player_points = 0
             if i > 0 and arena[i-1][j] == 2:
                 player_points += 2
             if i < row - 1 and arena[i+1][j] == 2:
                 player_points += 2
             if j > 0 and arena[i][j-1] == 2:
                 player_points += 2
             if j < col - 1 and arena[i][j+1] == 2:
                 player_points += 2
             toal_points += player_points
 if toal_points >= 10:
     print(f"Points Gained: {toal_points}. Your team has survived the game.")
 else:
     print(f"Points Gained: {toal_points}. Your team is out.")
