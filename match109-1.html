def mostWater(arr):
    max_water = float('-inf')
    left_height = 0
    right_height = len(arr) - 1
    while (left_height < right_height):
        width = right_height - left_height



        if (arr[left_height] < arr[right_height]):
            min_height = arr[left_height]
        else:                                            
            min_height = arr[right_height]
        container_area = min_height* width


        if (container_area > max_water):
            max_water = container_area
        if (arr[left_height] < arr[right_height]):
            left_height += 1
        else:
            right_height -= 1
    print(f"{max_water}")

    row = len(mat)
    col = len(mat[0])
    # print(row)
    # print(col)
    arr2 = np.empty((int(row/2), int(col/2)), dtype= int)
    new_rows , new_cols = arr2.shape
    
    for i in range(new_rows):
        for j in range(new_cols):
            arr2[i][j] = mat[2*i][2*j] + mat[2*i][2*j+1] + mat[2*i+1][2*j] + mat[2*i+1][2*j+1]

def play_game(arena):
    row = len(arena)
    col = len(arena[0])
    total_score = 0

    for i in range(row):
        for j in range(col):
            # print(arena[i][j] , end = " ")    
            
            if (arena[i][j] % 50 == 0) and (arena[i][j] != 0):


                count = 0    

                if i > 0 and arena[i-1][j] == 2:
                    count += 1
                if i < row-1 and arena[i+1][j] == 2:
                    count += 1
                if j > 0 and arena[i][j-1] == 2:
                    count += 1
