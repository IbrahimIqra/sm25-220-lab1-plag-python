def mostWater(arr):
    left = 0
    right = len(arr)-1
    max_area = 0


    while left < right:
        height = min(arr[left], arr[right])
        width = right - left
        area = height * width
        max_area = max(max_area, area)


        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1


    print(max_area)

def compress_matrix(mat):
  row,col= mat.shape
  arr= np.zeros(((row//2),(col//2)),dtype= int)
  arr1= np.zeros((col//2),dtype= int)
  count1=0


  for i in range(row):
    if i%2==0:
      count2=0
      sum=0
      for j in range(col):


        if j%2 == 0:
          sum = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]
          arr1[count2]=sum
          count2+=1
      arr[count1]=arr1
      arr1= np.zeros((col//2),dtype= int)
      count1+=1
  return arr

def play_game(arena):
  total_points = 0
  row, col = arena.shape


  for i in range(row):
    for j in range(col):


      if arena[i][j] % 50 == 0 and arena[i][j] != 0:


        if i - 1 >= 0 and arena[i - 1][j] == 2:
            total_points += 2


        if i + 1 < row and arena[i + 1][j] == 2:


            total_points += 2


        if j - 1 >= 0 and arena[i][j - 1] == 2:
            total_points += 2


        if j + 1 < col and arena[i][j + 1] == 2:
            total_points += 2


  if total_points >= 10:
      print(f"Points Gained: {total_points}. Your team has survived the game.")
