Task3- Matrix compressions

def compress_matrix(mat):
  row,col=matrix.shape
  new_sum=np.zeros((row//2,col//2),dtype=int)
  for i in range (0,row,2):
    for j in range(0,col,2):
      new_sum[i//2][j//2]=matrix[i][j]+matrix[i][j+1]+matrix[i+1][j]+matrix[i+1][j+1]
  return new_sum
