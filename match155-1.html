def mostWater(arr):
  #To Do
  left = 0
  right = len(arr) - 1
  area_max = 0

  while left < right:
    left_height = arr[left]
    right_height = arr[right]

    if left_height < right_height:
      current_height = left_height

    else:
      current_height = right_height

    width = right - left

    area = current_height * width
    
    if area > area_max:
      area_max = area

    if arr[left] < arr[right]:
      left += 1

    else:
      right -= 1

  print(area_max)

def compress_matrix(mat):
  #TO DO
    n_rows, n_cols = mat.shape
    new_rows = n_rows // 2
    new_cols = n_cols // 2

    compressed = np.zeros((new_rows, new_cols), dtype=int)

    for i in range(0, n_rows, 2):
        for j in range(0, n_cols, 2):
            block_sum = (
                mat[i][j]     + mat[i][j+1] +
                mat[i+1][j]   + mat[i+1][j+1]
            )
            compressed[i//2][j//2] = block_sum

    return compressed
