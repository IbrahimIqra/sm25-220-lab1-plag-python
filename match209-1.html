def mostWater(arr):
    n = len(arr)
    max_area = 0


    for i in range(n):
        for j in range(i + 1, n):


            if arr[i] < arr[j]:
                height = arr[i]
            else:
                height = arr[j]


            width = j - i
            area = height * width


            if area > max_area:
                max_area = area


    print(max_area)

    cols = len(arena[0])
    points_gained = 0

        for j in range(cols):
            if arena[i][j] > 0 and arena[i][j] % 50 == 0:
                if i > 0 and arena[i - 1][j] == 2:
                    points_gained += 2


                if i < rows - 1 and arena[i + 1][j] == 2:
