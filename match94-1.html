def mostWater(arr):

    n = len(arr)
    sum = 0
    for i in range(n):
        for j in range(i + 1, n):
            if arr[i] < arr[j]:
                h = arr[i]
            else:
                h = arr[j]
            width = j - i
            temp = h * width
            if temp > sum:
                sum = temp

def row_rotation(exam_week, seat_status):
  row,col = seat_status.shape
  for i in range(exam_week-1):
    temp_row = np.array(["  "]*row)
    for m in range(col):
      temp_row[m] = seat_status[row-1][m]
    for j in range(row-1,0,-1):
      for k in range(col):
        seat_status[j][k] = seat_status[j-1][k]
    for l in range(col):
      seat_status[0][l] = temp_row[l]


  for n in range(row):
    if seat_status[n][0] == 'Z':
      return n+1

def compress_matrix(mat):
    raw,col=mat.shape
    arr=np.zeros((raw//2,col//2),dtype=int)
    m,n=0,0
    for r in range(0,raw,2):
        for c in range(0,col,2):
            sum=0
            sum+=mat[r][c]+mat[r][c+1]+mat[r+1][c]+mat[r+1][c+1]
            arr[m][n]=sum
            n+=1
        m+=1
        n=0
    return arr
