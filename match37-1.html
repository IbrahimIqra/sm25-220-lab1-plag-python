def mostWater(arr):
  n = len(arr)
  max_area=0
  for i in range (0, n-1):
    for j in range (i+1, n):


      if arr[i] < arr[j]:
        h=arr[i]
      else:
        h = arr[j]
      w = j-i
      area = h*w
      if area > max_area:
        max_area = area
  print (max_area)

def play_game(arena):
  count=0
  row,col=arena.shape
  for r in range(row):
    for c in range(col):
      if arena[r][c]%50==0 and arena[r][c]!=0:
         if r>0 and arena[r-1][c]==2:
            count+=1
         if r<row-1 and arena[r+1][c]==2:
            count+=1
         if c>0 and arena[r][c-1]==2:
            count+=1
         if c<col-1 and arena[r][c+1]==2:
            count+=1
  points=count*2
  if points<10:
    print(f'Points Gained: {points}. Your team is out.')
  else:
    print(f'Points Gained: {points}. Your team has survived the game.')
