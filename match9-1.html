def play_game(arena):
  row,col=arena.shape
  p=0
  for i in range(row):
    for j in range(col):
      if arena[i][j]%50==0 and arena[i][j]!=0:
        if 0<i:
          if arena[i-1][j]==2:
            p+=2
        if i<row-1:
          if arena[i+1][j]==2:
            p+=2
        if 0<j:
          if arena[i][j-1]==2:
            p+=2
        if j<col-1:
          if arena[i][j+1]==2:
            p+=2

  if p<10:
    print(f"Points Gained: {p}. Your team is out.")
  else:
    print(f"Points Gained: {p}. Your team has survived the game.")
