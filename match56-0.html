def mostWater(arr):
  container=0
  for i in range(len(arr)):
    for j in range(len(arr)):
      width=abs(i-j)
      if arr[i]<arr[j]:
        height=arr[i]
      else:
        height=arr[j]
      tem=width*height
      if container<tem:
        container=tem
  return int(container)

def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0])

    new_rows = rows // 2
    new_cols = cols // 2
    compressed_mat = np.array([[0 for _ in range(new_cols)] for _ in range(new_rows)])


    for r_idx in range(new_rows):
        for c_idx in range(new_cols):
            block_sum = mat[2 * r_idx][2 * c_idx] + \
                        mat[2 * r_idx][2 * c_idx + 1] + \
                        mat[2 * r_idx + 1][2 * c_idx] + \
                        mat[2 * r_idx + 1][2 * c_idx + 1]

            compressed_mat[r_idx][c_idx] = block_sum



    return compressed_mat

    num_rows = len(arena)
    num_cols = len(arena[0])

    for r_idx in range(num_rows):
        for c_idx in range(num_cols):
            if arena[r_idx][c_idx] > 0 and arena[r_idx][c_idx] % 50 == 0:

                        total_points += 2
