import numpy as np


def mostWater(height):


    highest_water = 0
    for i in range(len(height)):  # i from 0 to 8
        for j in range(i + 1, len(height)):  # j from i+1 to 8
            width = j - i
            if height[i] <= height[j]:
                multiplication = height[i] * width
            else:
                multiplication = height[j] * width


            if multiplication > highest_water:
                highest_water = multiplication


    return int(highest_water)

import numpy as np


def compress_matrix(mat):


  row , col = len(mat)//2 , len(mat[0])//2
  compressed_array = np.zeros((row,col) , dtype =int)


  #first length of array = L . now, if i and j both %2 == 0 , loop should work.
  summation = 0
  for i in range (0,len(mat),2) :
     for j in range (0,len(mat[0]),2)  : #0,2,4
             summation = mat [i][j] + mat [i+1][j] +  mat [i][j+1] +  mat [i+1][j+1]
             compressed_array[i//2][j//2] = summation #0,0 0,1 0,2


  return compressed_array

def play_game(arena):
    survival_point = 0


    for i in range(len(arena)):
        for j in range(len(arena[0])):


            if arena[i][j] % 50 == 0 and arena[i][j] != 0:
               
                # middle area (has all 4 sides)
                if len(arena) - 1 > i > 0 and len(arena[0]) - 1 > j > 0:


                    if arena[i-1][j] == 2:
                        survival_point += 2
                    if arena[i+1][j] == 2:
                        survival_point += 2
                    if arena[i][j-1] == 2:


                        survival_point += 2
                    if arena[i][j+1] == 2:
                        survival_point += 2


                # corner cases


                elif i == 0 and j == 0:
                    if arena[i+1][j] == 2:
                        survival_point += 2
                    if arena[i][j+1] == 2:
                        survival_point += 2




                elif i == 0 and j == len(arena[0]) - 1:
                    if arena[i+1][j] == 2:
                        survival_point += 2
                    if arena[i][j-1] == 2:
                        survival_point += 2


                elif i == len(arena) - 1 and j == 0:


                    if arena[i-1][j] == 2:
                        survival_point += 2
                    if arena[i][j+1] == 2:
                        survival_point += 2




                elif i == len(arena) - 1 and j == len(arena[0]) - 1:
                    if arena[i-1][j] == 2:
                        survival_point += 2
                    if arena[i][j-1] == 2:
                        survival_point += 2


                # edge cases (non-corner)
                else:
                    if i == 0:
                        if i+1 < len(arena) and arena[i+1][j] == 2:
                            survival_point += 2
                        if j > 0 and arena[i][j-1] == 2:
                            survival_point += 2
                        if j+1 < len(arena[0]) and arena[i][j+1] == 2:
                            survival_point += 2


                    elif i == len(arena) - 1:
                        if arena[i-1][j] == 2:
                            survival_point += 2
                        if j > 0 and arena[i][j-1] == 2:
                            survival_point += 2
                        if j+1 < len(arena[0]) and arena[i][j+1] == 2:
                            survival_point += 2


                    elif j == 0:
                        if arena[i-1][j] == 2:
                            survival_point += 2


                        if arena[i+1][j] == 2:
                            survival_point += 2
                        if arena[i][j+1] == 2:
                            survival_point += 2


                    elif j == len(arena[0]) - 1:
                        if arena[i-1][j] == 2:
                            survival_point += 2
                        if arena[i+1][j] == 2:
                            survival_point += 2
                        if arena[i][j-1] == 2:
                            survival_point += 2


    print(f"Points Gained: {survival_point}.", end=" ")
