def mostWater(arr):
    hw=0
    rn=len(arr)-1
    tin_area=0
    while hw<rn:
        hgh=min(arr[hw],arr[rn])
        aln=rn-hw
        sim=hgh*aln
        tin_area=max(tin_area, sim)
        if arr[hw]<arr[rn]:
            hw+=1
        else:
            rn-=1
    print(tin_area)

  vf=np.array(seat_status[hf-1])
  for j in range(hf-1,0,1):
    seat_status[j]=seat_status[j-1]
  seat_status[0]=vf
  return seat_status
