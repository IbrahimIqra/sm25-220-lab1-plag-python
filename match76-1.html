def mostWater(arr):
    max_area = 0
    for left in range(len(arr)):
        for right in range(left + 1, len(arr)):
            height = 0


            if arr[left] < arr[right]:
                height = arr[left]
            else:
                height = arr[right]
            width = right - left
            area = height * width
            if area > max_area:
                max_area = area
    print(max_area)

def compress_matrix(mat):
    compress_matrix = np.zeros((2, 2), dtype=int)
    row_index = 0

    for i in range(len(mat)):
        col_index = 0
        for j in range(0, len(mat[0]), 2):
            compress_matrix[row_index//2][col_index] += mat[i][j]
            compress_matrix[row_index//2][col_index] += mat[i][j + 1]
            col_index += 1
        row_index += 1

    return compress_matrix

def play_game(arena):
    num_rows, num_cols = arena.shape
    points = 0
    for row in range(num_rows):
        for col in range(num_cols):
            if arena[row][col] % 50 == 0 and arena[row][col] != 0:
                if row + 1 < num_rows and arena[row + 1][col] == 2:
                    points += 2


                if row - 1 >= 0 and arena[row - 1][col] == 2:
                    points += 2
                if col + 1 < num_cols and arena[row][col + 1] == 2:
                    points += 2
                if col - 1 >= 0 and arena[row][col - 1] == 2:
                    points += 2
    if points < 10:
        print(f'Points Gained: {points}. Your team is out.')
    if points>=10:
        print(f'Points Gained: {points}. Your team has survived the game.')
