def mostWater(heights):
    maxArea = 0
    Rock = 0
    Cena = len(heights) - 1


    while Rock < Cena:


        if heights[Rock] < heights[Cena]:
            KevinHeight = heights[Rock]
        else:
            KevinHeight = heights[Cena]


        Kevin = Cena - Rock


        KevinArea = KevinHeight * Kevin


        if KevinArea > maxArea:
            maxArea = KevinArea


        if heights[Rock] < heights[Cena]:
            Rock += 1
        else:
            Cena -= 1


    print(maxArea)

def compress_matrix(matrix):
    row = len(matrix)
    col = len(matrix[0])


    new_rows = row // 2
    new_cols = col // 2


    new_array = []
    for _ in range(new_rows):
        new_array.append([0]*new_cols)


    for i in range(0, row, 2):
        for j in range(0, col, 2):
            s = matrix[i][j] + matrix[i][j+1] + matrix[i+1][j] + matrix[i+1][j+1]
            new_array[i//2][j//2] = s


    import numpy as np
    return np.array(new_array)

def play_game(arena):
    rows = len(arena)
    cols = len(arena[0])
    points_gained = 0


    for i in range(rows):
        for j in range(cols):
            now = arena[i][j]
            if now != 0 and now % 50 == 0:


                points = 0


                if i > 0 and arena[i - 1][j] == 2:
                    points += 1


                if i < rows - 1 and arena[i + 1][j] == 2:
                    points += 1


                if j > 0 and arena[i][j - 1] == 2:
                    points += 1


                if j < cols - 1 and arena[i][j + 1] == 2:
                    points += 1
                points_gained += points * 2


    if points_gained >= 10:
        print("Points Gained: " + str(points_gained) + ". Your team has survived the game.")
    else:
        print("Points Gained: " + str(points_gained) + ". Your team is out.")
