def compress_matrix(mat):
  #TO DO
  row,col=mat.shape
  new_matrix=np.zeros((row//2,col//2),dtype=int)


  for i in range(0, row, 2):


    for j in range(0,col,2):
      new_matrix[i//2][j//2]=mat[i][j]+mat[i+1][j]+mat[i][j+1]+mat[i+1][j+1]


  return new_matrix

def play_game(arena):
  #TO DO
  r,c=arena.shape
  total_point=0


  for m in range(r):




    for n in range(c):




      if arena[m][n]>0 and arena[m][n]%50==0:


        player_point=0


        if m>0 and arena[m-1][n]==2:


          player_point+=1


        if m<r-1 and arena[m+1][n]==2:




          player_point+=1


        if n>0 and arena[m][n-1]==2:


          player_point+=1


        if n<c-1 and arena[m][n+1]==2:

        total_point+=player_point*2


  if total_point>=10:
