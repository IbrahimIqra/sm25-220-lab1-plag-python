import numpy as np




def mostWater(arr):
  max = 0
  l = 0
  r = len(arr) - 1


  while l < r:
    hgt = min(arr[l], arr[r])
    wdt = r - l
    area = hgt * wdt


    if area > max:
      max = area


    if arr[l] < arr[r]:
      l += 1
    else:
      r -= 1


  print(max)

def compress_matrix(mat):
  row, col = mat.shape
  n_row = row // 2
  n_col = col // 2
  com = np.zeros((n_row, n_col), dtype=int)




  for i in range(0, row, 2):
    for j in range(0, col, 2):
      block_sum = (
        mat[i][j] +
        mat[i][j+1] +
        mat[i+1][j] +
        mat[i+1][j+1]
      )
      com[i//2][j//2] = block_sum


  return com

def play_game(arena):
  row, col = arena.shape
  t_points = 0


  for i in range(row):
    for j in range(col):
      if arena[i][j] != 0 and arena[i][j] % 50 == 0:


        count = 0
        if i > 0 and arena[i-1][j] == 2:
          count += 1
        if i < row - 1 and arena[i+1][j] == 2:
          count += 1
        if j > 0 and arena[i][j-1] == 2:
          count += 1
        if j < col - 1 and arena[i][j+1] == 2:
          count += 1
        t_points += count * 2


  if t_points >= 10:
    print(f"Points Gained: {t_points}. Your team has survived the game.")
