def mostWater(arr):




 left = 0
 right = len(arr) - 1
 max_wate = 0


 while left < right:
   width = right - left
   height = min(arr[left], arr[right])
   max_wate = max(max_wate, width * height)


   if arr[left] < arr[right]:
     left += 1
   else:
     right -= 1

def compress_matrix(mat):
 row, col = mat.shape
 compresed = np.zeros((row // 2, col // 2), dtype=int)


 for i in range(0, row, 2):
   for j in range(0, col, 2):
     block_su = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]


     compresed[i//2][j//2] = block_su


 return compresed

def play_game(arena):
 rows, cols = arena.shape
 points = 0


 for i in range(rows):
   for j in range(cols):
     if arena[i][j] % 50 == 0 and arena[i][j] != 0:
      


       if i - 1 >= 0 and arena[i - 1][j] == 2:
         points += 2
      
       if i + 1 < rows and arena[i + 1][j] == 2:
         points += 2
     
       if j - 1 >= 0 and arena[i][j - 1] == 2:
         points += 2
      
       if j + 1 < cols and arena[i][j + 1] == 2:
         points += 2


 if points >= 10:
