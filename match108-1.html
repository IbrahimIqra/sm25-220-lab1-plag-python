def compress_matrix(mat):
  row,col = mat.shape
  result = np.zeros(((row//2),(col//2)),dtype= object)
  ro = 0
  cl = 0
  for i in range(0,row,2):
    for j in range(0,col,2):
      sum = 0


      sum+= mat[i][j]+ mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
      result[ro][cl] = sum
      cl+=1  #col 2 te 22 boshbe
    ro+=1  #row 2 te 2 boshbe
