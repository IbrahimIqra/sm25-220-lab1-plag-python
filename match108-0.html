def mostWater(arr):
  row=len(arr)
  col=len(arr)
  max_a=0
  for i in range(row):
    for j in range(col):
      width=j-i
      if arr[i]<arr[j]:
        height=arr[i]
      else:
        height=arr[j]
      area=width*height
      if area>max_a:
        max_a=area
  print(max_a)

def play_game(arena):
  row=len(arena)
  col=len(arena[0])
  total=0
  for i in range(row):
    for j in range(col):


      if arena[i][j]!=0 and arena[i][j]%50==0:
        if j+1<col and arena[i][j+1]==2:
          total+=2
        if j-1>=0 and arena[i][j-1]==2:
          total+=2
        if i+1<row and arena[i+1][j]==2:
          total+=2
        if i-1>=0 and arena[i-1][j]==2:
          total+=2
  if total<10:
    print(f"Points Gained: {total}. Your team is out.")
  else:
    print(f"Points Gained: {total}. Your team has survived the game.")
