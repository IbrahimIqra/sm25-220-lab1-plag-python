def mostWater(arr):
  
  max_water = 0
  left = 0
  right = len(arr) - 1

  while left < right:
    height_left = arr[left]
    height_right = arr[right]
    min_height = min(height_left, height_right)
    width = right - left
    current_water = min_height * width
    max_water = max(max_water, current_water)

    if height_left < height_right:
      left += 1
    else:
      right -= 1

  print(max_water)

def compress_matrix(mat):
    rows = len(mat)


    cols = len(mat[0])
    result = np.zeros((rows // 2, cols // 2), dtype=int)
    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            result[i // 2][j // 2] = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
    return result



def play_game(arena):
    rows, cols = len(arena), len(arena[0])
    total_points = 0

    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:


                if i > 0 and arena[i - 1][j] == 2:
                    total_points += 2
                if i < rows - 1 and arena[i + 1][j] == 2:
                    total_points += 2
                if j > 0 and arena[i][j - 1] == 2:
                    total_points += 2
                if j < cols - 1 and arena[i][j + 1] == 2:
                    total_points += 2

    if total_points >= 10:
        print(f"Points Gained: {total_points}. Your team has survived the game.")
    else:
        print(f"Points Gained: {total_points}. Your team is out.")
