def compress_matrix(mat):
   row,col=mat.shape
   new = np.zeros((row//2, col//2), dtype=int)


   for i in range(0, row, 2):
       for j in range(0, col, 2):
           sum = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]


           new[i//2][j//2] = sum


   return new

def play_game(arena):
   rows, col = arena.shape
   tp = 0
   for i in range(rows):
       for j in range(col):
           if arena[i][j] % 50==0 and arena[i][j] != 0:
               point = 0


               if i-1 >= 0 and arena[i-1][j]==2:
                   point+=2
               if i+1 < rows and arena[i+1][j]==2:
                   point+=2
               if j-1 >= 0 and arena[i][j-1]==2:
                   point+=2
               if j+1 < col and arena[i][j+1]==2:
                   point+=2


               tp+=point


   if tp>=10:
       print(f"Points Gained: {tp}. Your team has survived the game.")
   else:
       print(f"Points Gained: {tp}. Your team is out.")
