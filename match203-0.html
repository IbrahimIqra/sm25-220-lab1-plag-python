def mostWater(arr):
  l=0
  r= len(arr)-1
  capacity=0
  while l<r:
    breath = r-1
    height= min(arr[l], arr[r])


    area = height*breath
    if area>capacity:
      capacity=area
    if arr[l]<arr[r]:
      l+=1
    else:
      r-=1
  print(capacity)
