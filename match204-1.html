def mostWater(arr):


    l = 0

    r = len(arr) - 1

    max_area = 0


    while l < r:

        hei = min(arr[l], arr[r])

        wid = r - l

        max_area = max(max_area, hei * wid)

        if arr[l] < arr[r]:

            l += 1

        else:

            r -= 1


    print(max_area)

def row_rotation(exam_week, seat_status):


    row, column = seat_status.shape

    steps = exam_week % row

    r_seat_status = np.roll(seat_status, -steps, axis=0)

    for i in range(row):

        if "AA" in r_seat_status[i]:

            row_no = i + 1

            break

    print_matrix(r_seat_status)

    return row_no


def print_matrix(m):


    row, column = m.shape

    for i in range(row):

        c = 1

        print('|', end='')

        for j in range(column):

            c += 1

            if len(str(m[i][j])) == 1:

                print(' ', m[i][j], end='  |')

                c += 6

            else:

                print(' ', m[i][j], end=' |')

                c += 6

        print()

        print('-' * (c - column))

def compress_matrix(mat):


    rows, columns = mat.shape



    new_c_m = np.zeros((rows // 2, columns // 2), dtype = int)

    for i in range(0, rows, 2):

        for j in range(0, columns, 2):

            block_sum = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]

            new_c_m[i // 2][j // 2] = block_sum

    return new_c_m

def play_game(arena):


    rows = len(arena)

    columns = len(arena[0])

    t = 0

    for i in range(rows):

        for j in range(columns):

            if arena[i][j] % 50 == 0 and arena[i][j] != 0:



                if i > 0 and arena[i - 1][j] == 2:

                    t += 2

                if i < rows - 1 and arena[i + 1][j] == 2:

                    t += 2

                if j > 0 and arena[i][j - 1] == 2:

                  if j < columns - 1 and arena[i][j + 1] == 2:

                    t += 2

    print(f"Points Gained: {t}.", end=" ")

    if t >= 10:

        print("Your team has survived the game.")

    else:

        print("Your team is out.")
