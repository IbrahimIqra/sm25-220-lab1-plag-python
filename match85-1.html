def mostWater(arr):


    left, right = 0, len(arr) - 1
    area = 0


    while left < right:
      width = right - left
      height = min(arr[left], arr[right])
      area = max(area, height * width)






      if arr[left] < arr[right]:
          left += 1
      else:
          right -= 1


    print(area)

def compress_matrix(mat):
  #TO DO


  r, c = mat.shape
  new_c = c // 2
  new_r = r // 2
  new_p = np.zeros((new_r, new_c), dtype=int)
  for i in range(0, r, 2):
    for j in range(0, c, 2):
      s = (mat[i, j] + mat[i+1, j] + mat[i, j+1] + mat[i+1, j+1])
      new_p[i//2, j//2] = s


  return new_p

def play_game(arena):
  s = 0
  r, c = arena.shape


  way = [(-1, 0), (1, 0), (0, -1), (0, 1)]
  for i in range(r):
    for j in range(c):
      if arena[i][j] % 50 == 0 and arena[i][j] != 0:
        ni = 0
        nj = 0


        for k in way:
          ni, nj = i + k[0], j + k[1]
          if 0 <= ni < r and 0 <= nj < c and arena[ni][nj] == 2:
            s += 2


  print(f'Points Gained: {s}.', end=' ')
  if s >= 10:
    print('Survived the game.')
  else:
    print('Your team is out.')
