def mostWater(arr):
    x = 0
    y = len(arr) - 1
    max_area = 0


    while x < y:
        width = y - x
        if arr[x] < arr[y]:
            height = arr[x]
        else:
            height = arr[y]

        curr_area = width * height

        if curr_area > max_area:
            max_area = curr_area

        if arr[x] < arr[y]:
            x += 1
        else:
            y -= 1

    print(max_area)

def compress_matrix(mat):
  
  row,col = mat.shape
  compressed_matrix1=np.zeros((row//2,col//2),dtype=int)
  for i in range(0,row,2):
    for j in range(0,col,2):
      compressed_matrix1[i//2,j//2]=(mat[i,j]+mat[i,j+1]+mat[i+1,j]+mat[i+1,j+1])
  return compressed_matrix1
def print_matrix(matrix):
  for row in matrix:
    print(" | ", end=" ")
    for col in row:
      print(col,end=" | ")
    print("")
  print("-"*(len(matrix[0]*4+1)))

def play_game(arena):
  

    total_points = 0
    for i in range(len(arena)):
        for j in range(len(arena[0])):


            if arena[i][j]>0 and arena[i][j] % 50 == 0:
                points = 0

                if i > 0 and arena[i - 1][j] == 2:
                    points += 1

                if i < len(arena) - 1 and arena[i + 1][j] == 2:


                    points += 1

                if j > 0 and arena[i][j - 1] == 2:
                    points += 1

                if j < len(arena[0]) - 1 and arena[i][j + 1] == 2:


                    points += 1
                total_points += points *2

    if total_points >= 10:
        print (f"Points Gained: {total_points}. Your team has survived the game.")
    else:
        print (f"Points Gained: {total_points}. Your team is out.")
