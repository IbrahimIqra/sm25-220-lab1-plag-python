def mostWater(arr):
        area = 0
        a = 0
        b = len(arr) - 1

        for i in range(len(arr)):
            lent = b - a
            if arr[a] <= arr[b]:
                if area < arr[a] * lent:
                    area = arr[a] * lent
                a += 1
            else:
                if area < arr[b] * lent:
                    area = arr[b] * lent
                b -= 1

        return int(area)

def compress_matrix(mat):
   rows, cols= mat.shape


   new_rows=rows//2
   new_cols=cols//2
   result =np.zeros((new_rows,new_cols),dtype=int)

   for row in range(0,rows,2):
     for col in range(0,cols,2):
        temp=mat[row][col]+mat[row+1][col]+mat[row][col+1]+mat[row+1][col+1]


        result[row//2][col//2]=temp

   return result

def play_game(arena):
    rows,cols=arena.shape
    points=0


    for i in range(rows):
        for j in range(cols):
            player_points=0
            if arena[i][j] % 50 == 0 and arena[i][j] !=0 and arena[i][j] != 1 :


                if i > 0 and arena[i-1][j] == 2: #up
                    player_points += 2
                if i < rows - 1 and arena[i+1][j] == 2: #down
                    player_points += 2
                if j > 0 and arena[i][j-1] == 2:    #left
                    player_points += 2
                if j < cols - 1 and arena[i][j+1] == 2:   #r8
                    player_points += 2

            points += player_points

    if points >= 10:
        print(f"Points Gained: {points}. Your team has survived the game.")
    else:
        print(f"Points Gained: {points}. Your team is out.")
