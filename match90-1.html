def mostWater(arr):
  max=0
  for i in range(len(arr)-1,-1,-1):
    for j in range(i,-1,-1):
      if arr[j]>=arr[i]:
        if arr[i]*(i-j)>=max:
          max=arr[i]*(i-j)
  print(max)

def compress_matrix(mat):
  rows,columns=mat.shape
  new_arr=np.zeros((rows//2,columns//2),dtype=int)
  output_i=0
  for i in range(0,rows,2):
    output_j=0
    for j in range(0,columns,2):
      new_arr[output_i][output_j]=mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
      output_j+=1
    output_i+=1
  return new_arr
