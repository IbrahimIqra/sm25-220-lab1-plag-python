def mostWater(arr):
  c_left=0
  c_right=len(arr)-1
  max_area=0

  for i in range( len(arr)-1):
    if arr[c_left]< arr[c_right]:
      h= arr[c_left]

    else:


      h= arr[c_right]

    w= c_right - c_left
    new_area= h * w
    if new_area> max_area:
      max_area= new_area

    if arr[c_left]< arr[c_right]:
      c_left= c_left +1

    else:
      c_right= c_right -1 

  print ( max_area)

def play_game(arena):
    rows, cols = arena.shape


    t_point = 0

    for i in range(rows):
      for j in range(cols):
          if arena[i][j] % 50 == 0 and arena[i][j] != 0:
                
              points = 0
                


              if i > 0 and arena[i-1][j] == 2:
                  points += 2
                
              if i < rows - 1 and arena[i+1][j] == 2:
                  points += 2
                
              if j > 0 and arena[i][j-1] == 2:
                  points += 2
                
              if j < cols - 1 and arena[i][j+1] == 2:
                  points += 2
                
              t_point += points
    
    if t_point >= 10:
        print(f"Points Gained: {t_point}. Your team has survived the game.")
    else:
        print(f"Points Gained: {t_point}. Your team is out.")
