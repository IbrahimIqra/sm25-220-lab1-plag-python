import numpy as np


def mostWater(arr):
  left_pointer=0
  right_pointer=len(arr)-1
  max_area=0


  while left_pointer<right_pointer:


    if arr[left_pointer]<arr[right_pointer]:
      current_height=arr[left_pointer]
     
    else:
      current_height=arr[right_pointer]
     


    width=right_pointer-left_pointer
    current_area=width*current_height


    if max_area<=current_area:
      max_area=current_area
    else:
      pass


    if arr[left_pointer]<arr[right_pointer]:
      left_pointer+=1
    else:
      right_pointer-=1


  print(max_area)

import numpy as np


def compress_matrix(mat):
  row=len(mat)
  column=len(mat[0])
  n_row= row//2
  n_column= column//2


  new_mat=np.empty((n_row, n_column), dtype=int)
 
  for r in range(0, row,2):
    for c in range(0, column, 2):
      sum=(mat[r][c]+
           mat[r][c+1]+
           mat[r+1][c]+
           mat[r+1][c+1])


      new_mat[r//2][c//2]=sum
 
  return new_mat

def play_game(arena):


  rows,cols = arena.shape
  points =0
  for r in range(rows):
    for c in range(cols):
      if (arena[r][c])>0 and (arena[r][c])%50==0:




        if c-1>-1 and arena[r][c-1]==2:
          points+=1
        if c+1<cols and arena[r][c+1]==2:
          points+=1
        if r-1>-1 and arena[r-1][c]==2:


          points+=1
        if r+1<rows and arena[r+1][c]==2:
          points+=1


  total_points = points*2
  if total_points>=10:
    print(f'Points gained:{total_points}. Your team has survived the game.')
  else:
    print(f'Points gained:{total_points}. Your team is out.')
  return total_points
