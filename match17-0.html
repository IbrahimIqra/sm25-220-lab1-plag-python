def mostWater(arr):
  # To Do
  left_pos = 0
  right_pos = len(arr) - 1
  total_area = 0


  while left_pos < right_pos:
    if arr[left_pos] < arr[right_pos]:
      height = arr[left_pos]
    else:
      height = arr[right_pos]
    width = right_pos - left_pos
    areas = height * width
    if areas > total_area:
      total_area = areas
    if arr[left_pos] < arr[right_pos]:
      left_pos += 1
    else:
      right_pos -= 1
  print(total_area)

def compress_matrix(mat):
  #TO DO
  rows = len(mat)
  colomn = len(mat[0])
  new_rows = rows // 2
  new_colomn = colomn // 2
  zip = np.zeros((new_rows, new_colomn), dtype=int)
  for i in range(0, rows, 2):
      for j in range(0, colomn, 2):
          block_sum = (
              mat[i][j] +
              mat[i][j+1] +
              mat[i+1][j] +
              mat[i+1][j+1]
              )
          zip[i//2][j//2] = block_sum
  return zip

def play_game(arena):
  points = 0
  rows = len(arena)
  colomn = len(arena[0])
  directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
  for i in range(rows):
    for j in range(colomn):
      if arena[i][j] != 0 and arena[i][j] % 50 == 0:
        for axis in directions:
          x = axis[0]
          y = axis[1]
          p = i + x
          k = j + y
          if p >= 0 and p < rows and k >= 0 and k < colomn:
            if arena[p][k] == 2:
              points += 2
  if points >= 10:
    print(f"Points Gained: {points}. Your team has survived the game.")
  else:
    print(f"Points Gained: {points}. Your team is out.")
