def mostWater(arr):
   left = 0
   right = len(arr) - 1
   max_area = 0


   while left < right:
       height = min(arr[left], arr[right])
       width = right - left
       area = height * width
       if area > max_area:
           max_area = area
       if arr[left] < arr[right]:
           left += 1
       else:
           right -= 1


   print(max_area)

def row_rotation(exam_week, seat_status):
   total_rows = len(seat_status)


   shift = exam_week - 1


   rotated = []
   for i in range(total_rows):
       rotated_index = (i - shift) % total_rows
       rotated.append(seat_status[rotated_index])


   for row in rotated:
       row_str = ""
       for i in range(len(row)):
           row_str += str(row[i])
           if i != len(row) - 1:
               row_str += " | "
       print("| " + row_str + " |")


   for i in range(total_rows):
       row = rotated[i]
       for j in range(len(row)):
           if row[j] == "AA":
               return i + 1

def compress_matrix(mat):
   rows = len(mat)
   cols = len(mat[0])
   new_rows = rows // 2
   new_cols = cols // 2


   compressed = np.zeros((new_rows, new_cols), dtype=int)


   for i in range(0, rows, 2):
       for j in range(0, cols, 2):
           block_sum = (
               mat[i][j] + mat[i][j+1] +
               mat[i+1][j] + mat[i+1][j+1]
           )
           compressed[i//2][j//2] = block_sum


   return compressed

def play_game(arena):
   rows = len(arena)
   cols = len(arena[0])


   total_twos = 0


   directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]


   for r in range(rows):
       for c in range(cols):
           cell = arena[r][c]
           if cell != 0 and cell % 50 == 0:
               count_adjacent_twos = 0


               for dr, dc in directions:
                   nr, nc = r + dr, c + dc
                   if 0 <= nr < rows and 0 <= nc < cols:
                       if arena[nr][nc] == 2:
                           count_adjacent_twos += 1
               total_twos += count_adjacent_twos


   points = total_twos * 2


   if points >= 10:
       print(f"Points Gained: {points}. Your team has survived the game.")
   else:
       print(f"Points Gained: {points}. Your team is out.")
