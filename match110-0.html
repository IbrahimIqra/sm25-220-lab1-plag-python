def mostWater(height):
    left= 0
    right= len(height) - 1
    max_area= 0

    while left<right:


        if height[left]< height[right]:
            h= height[left]
        else:
            h= height[right]

        w= right - left
        
        area= h * w
        
        if area > max_area:
            max_area= area
        
        if height[left] < height[right]:
            left+= 1
        else:
            right-= 1
print(max_area)

def row_rotation(exam_week, seat_status):
    num_rows = len(seat_status)
    rotations = exam_week - 1

    for _ in range(rotations):
        last_row= seat_status[-1]
        for i in range(num_rows - 1, 0, -1):
            seat_status[i]= seat_status[i - 1]

        seat_status[0]= last_row

    print_matrix(seat_status)

    for i in range(num_rows):
        for j in range(len(seat_status[0])):
            if seat_status[i][j]== 'AA':
                return i + 1

def play_game(arena):
    rows= len(arena)
    cols= len(arena[0])
    total_points= 0

    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50== 0 and arena[i][j]!= 0:
                points = 0



                if i > 0 and arena[i-1][j]== 2:
                    points+= 1
                if i < rows - 1 and arena[i+1][j]== 2:
                    points+= 1
                if j > 0 and arena[i][j-1]== 2:
                    points += 1
                if j < cols- 1 and arena[i][j+1]== 2:
                    points+= 1

                total_points+= points

    total_points*= 2

    if total_points >= 10:
        print(f'Points Gained: {total_points}. Your team has survived the game.')
    else:
        print(f'Points Gained: {total_points}. Your team is out.')
