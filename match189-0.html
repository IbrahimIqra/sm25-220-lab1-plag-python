def row_rotation(exam_week, seat_status):
    row=len(seat_status)
    col=len(seat_status[0])
    new=np.zeros((row,col),dtype=int)


    for i in range(row):
        for j in range(col):
            if seat_status[i][j]=='AA':
                new[i][j]=seat_status[(i+exam_week)%row][j]
                return new

def compress_matrix(mat):
    row=len(mat)
    col=len(mat[0])
    new=np.zeros(row//2,col//2)


    for i in range(row):
        for j in range(col):


            sum=mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[j+1][1]
            new=sum
            return new
