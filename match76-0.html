def compress_matrix(mat):
  row,col=mat.shape
  arr1= np.zeros((row//2, col//2),dtype = int)
  for r in range(0,row,2):
    for c in range(0,col, 2):
      arr1[r//2][c//2]=mat[r][c]+mat[r+1][c]+mat[r][c+1]+mat[r+1][c+1]
  return arr1

def play_game(arena):
  row,col=arena.shape
  PT= 0
  for r in range(row):
    for c in range(col):
      if((arena[r][c]%50==0) and (arena[r][c]!= 0)):
        if((r-1>=0) and (arena[r-1][c]==2)):


          PT+=2
        if((r+1<row) and (arena[r+1][c]==2)):
          PT+=2
        if((c-1>=0) and (arena[r][c-1]==2)):
          PT+=2
        if((c+1<col) and (arena[r][c+1]==2)):
          PT+=2
  if(PT<10):
    print(f"Points Gained:{PT}.Your team is out.")
  else:
    print(f"Points Gained:{PT}.Your team has survived the game.")
