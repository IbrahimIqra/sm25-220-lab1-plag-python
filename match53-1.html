import numpy as np


def compress_matrix(mat):
    rows, cols = len(mat), len(mat[0])


    c_rows, c_cols = rows // 2, cols // 2
    compressed = np.zeros((c_rows, c_cols), dtype=int)
    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            block_sum = (mat[i][j] + mat[i][j + 1] +mat[i + 1][j] + mat[i + 1][j + 1])


            compressed[i // 2][j // 2] = block_sum
    return compressed
def print_matrix(matrix):
    for row in matrix:
        print("|", end=" ")
        for element in row:
            print(f"{element:3}", end=" | ")
        print("\n" + "-" * (len(row) * 5 + 1))

def play_game(arena):
    row,col = arena.shape
    sum = 0
    for i in range(len(arena)):
        for j in range(len(arena[i])):


            if arena[i][j]%50 == 0:
                if i+1<row and arena[i+1][j] == 2:
                    sum += 2
                if i-1<-1 and arena[i-1][j] == 2:
                    sum += 2
                if j+1<col and arena[i][j+1] == 2:
                    sum += 2
                if j-1>col and arena[i][j-1] == 2:
                    sum += 2
    if sum < 10:
        print(f'points gained: {sum}. Your team is out.')
    else:
        print(f'points gained: {sum}. Your team has survived the game.')
