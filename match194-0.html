def mostWater(arr):
  pointer1 = 0
  pointer2 = len(arr)-1
  max_area = 0
  while pointer1 < pointer2:


    if arr[pointer1] < arr[pointer2]:
      height = arr[pointer1]
    else:
      height = arr[pointer2]
    width = pointer2 - pointer1
    area = height * width
    if area > max_area:
      max_area = area
    if arr[pointer1] < arr[pointer2]:
      pointer1 += 1
    else:
      pointer2 -= 1
  print(max_area)

def row_rotation(exam_week, seat_status):
  row,col = len(seat_status),len(seat_status[0])


  for i in range(exam_week-1):
    temp_row=np.array([' ']*row)
    for m in range(col):
      temp_row[m]=seat_status[row-1][m]
    for j in range(row-1,0,-1):
      for k in range(col):
        seat_status[j][k]=seat_status[j-1][k]
    for l in range(col):
      seat_status[0][l]=temp_row[l]

  for n in range(row):
    if seat_status[n][0]=='Z':
      return n+1

def compress_matrix(mat):
  row,col = len(mat),len(mat[0])
  result = np.zeros((row//2,col//2),dtype=int)
  count_row = -1
  count_col = -1
  for i in range(0,row+1//2,2):
    count_row+=1
    for j in range(0,col+1//2,2):
      count_col+=1


      result[count_row][count_col] = mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
      if count_col == 1:
        count_col = -1
  return result

  row,col = len(arena),len(arena[0])
  for i in range(row):


    for j in range(col):
      if arena[i][j]%50==0 and arena[i][j]!=0:
        if j+1<row-1 and arena[i][j+1]==2:
          point+=2
        if i-1>=0 and arena[i-1][j]==2:
          point+=2
        if i+1<col-1 and arena[i+1][j]==2:
