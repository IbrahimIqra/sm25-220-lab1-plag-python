import numpy as np
def mostWater(height):
  arr=height
  length = len(arr)
  right = length-1
  left = 0
  max=0
  while left<=right:


    if arr[left] < arr[right]:
      height= arr[left]
    else:
      height= arr[right]


    area = height*(right-left)


    if area > max:
      max = (area)  


    if arr[left]<arr[right]:
      left+=1
    else:
      right-=1


  return int(max)    #returns array64format without int

def reverse(matrix, start, end):
    mat=matrix
    while start < end:
       
        for col in range(len(mat[0])):
            temp = mat[start][col]
            mat[start][col] = mat[end][col]
            mat[end][col] = temp
        start += 1
        end -= 1

import numpy as np
def compress_matrix(mat):
    row = len(mat)
    col = len(mat[0])
    newROW = row // 2
    newCOL = col // 2
    comps = np.zeros((newROW, newCOL), dtype="int")
 
    i = 0
    while i < row:
        j = 0
        while j < col:


            sum = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            comps[i//2][j//2] = sum     #sum in compressed
            j += 2
        i += 2
       
    return comps
