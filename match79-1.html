def compress_matrix(mat):
    #TO DO
    r,c = mat.shape
    new_arr = np.zeros((r//2,c//2), dtype="int")
    row= 0
    col = 0


    for i in range(0,r,2):
      for j in range(0,c,2):


        sum = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
        new_arr[row][col] = sum


        col +=1


      col = 0
      row += 1


    return new_arr
