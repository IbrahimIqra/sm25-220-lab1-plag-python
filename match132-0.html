def mostWater(arr):
    left = 0
    right = len(arr) - 1
    max_area = 0

    while left < right:
        height = min(arr[left], arr[right])
        width = right - left
        area = height * width
        max_area = max(max_area, area)

        # Move the pointer with the smaller height
        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1

    print(max_area)

def compress_matrix(mat):
    row,col = mat.shape
    r = row//2
    c = col//2


    arr = np.array([[0]*c]*r)
    sum = 0
    rw = 0
    cl = 0
    for i in range(0,row,2):
      for j in range(0,col,2):
        sum = mat[i][j]+mat[i+1][j]+mat[i][j+1]+mat[i+1][j+1]
        arr[rw][cl] = sum
        cl+=1
      rw+=1
      cl=0
      sum=0
    return arr
