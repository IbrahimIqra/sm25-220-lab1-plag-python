def mostWater(arr):
  left = 0
  right= len(arr)-1
  max = 0
  while left < right:
    if arr[right] > arr[left]:
      height = arr[left]
      width = right - left
      left +=1
    else:
      height = arr[right]
      width = right - left
      right -=1
    
    if height*width > max:
        max = height*width
  return max

def compress_matrix(mat):
  row, col = mat.shape


  new_mat = np.zeros((row//2, col//2), dtype= int)

  for i in range(0,row, 2):
    for j in range(0, col, 2):
      sum = mat[i][j]+ mat[i][j+1]+ mat[i+1][j] + mat[i+1][j+1]
      new_mat[i//2][j//2] = sum

  return new_mat
