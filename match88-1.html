def mostWater(arr):
  left=0
  right= len(arr)-1
  maximum=0


  while left<right:
    width=right - left
    current_height=min(arr[left], arr[right])
    current_area=width*current_height


    maximum= max(maximum, current_area)


    if arr[left]<arr[right]:
      left+=1
    else:
      right-=1
  return maximum

def compress_matrix(mat):
  row=len(mat)
  cols=len(mat[0])




  compressed_rows=row//2
  compressed_cols=cols//2




  compressed_matrix=np.zeros((compressed_rows,compressed_cols),dtype=int)




  for i in range(0, row, 2):
    for j in range(0, cols, 2):
      block_sum=(mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1])
      compressed_matrix[i//2][j//2]=block_sum




  return compressed_matrix
