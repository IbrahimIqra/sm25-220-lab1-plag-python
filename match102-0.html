import numpy as np
def mostWater(arr):
    left = 0
    right = len(arr) - 1
    max_area= 0
    while left < right:
        height = min(arr[left], arr[right])
        breadth = right - left
        area = height * breadth
        max_area = max(max_area, area)
        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1
    print(max_area)

import numpy as np
def compress_matrix(mat):
    compressed = [[0, 0], [0, 0]]
    i = 0
    while i < 4:
        j = 0
        row = [0, 0]


        while j < 4:  
            sum_block = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            row[j//2] = sum_block  
            j += 2  
        compressed[i//2] = row
        i += 2  
    return np.array(compressed)
