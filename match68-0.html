import numpy as np


def mostWater(arr):
    n = len(arr)
    left = 0
    right = n - 1
    max_area = 0


    while left < right:
        h_left = arr[left]
        h_right = arr[right]
        
        height = h_left if h_left < h_right else h_right
        width = right - left
        area = height * width


        if area > max_area:
            max_area = area


        if h_left < h_right:
            left += 1
        else:
            right -= 1


    print(max_area)

def row_rotation(exam_week, seat_status):
    rows = len(seat_status)
    cols = len(seat_status[0])


    rotation_count = exam_week - 1  


    rot = 0
    while rot < rotation_count:
        temp_row = seat_status[rows - 1].copy()


        i = rows - 1
        while i > 0:
            seat_status[i] = seat_status[i - 1]
            i -= 1


        seat_status[0] = temp_row


        rot += 1


    aa_row = -1
    i = 0
    while i < rows:
        j = 0
        while j < cols:
            if seat_status[i][j] == "AA":
                aa_row = i + 1
                break
            j += 1
        if aa_row != -1:
            break
        i += 1


    print_matrix(seat_status)


    return aa_row

import numpy as np


def print_matrix(matrix):
    rows = len(matrix)
    cols = len(matrix[0])
    i = 0
    while i < rows:
        j = 0
        while j < cols:
            print(f"| {str(matrix[i][j]).rjust(3)} ", end="")
            j += 1
        print("|")
        print("-" * (cols * 6))
        i += 1




def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0])
    
    new_rows = rows // 2
    new_cols = cols // 2
    
    compressed = np.zeros((new_rows, new_cols), dtype=int)
    
    i = 0
    while i < rows:
        j = 0
        while j < cols:
            sum_block = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            compressed[i//2][j//2] = sum_block
            j += 2
        i += 2
    
    return compressed
