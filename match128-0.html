def mostWater(height):
  total_area = 0
  left = 0
  right = len(height) - 1
  while left < right:


    if height[left] > height[right]:
        h = height[right]
    else:
        h = height[left]

    width = right - left
    area = width * h

    if area > total_area:
        total_area = area

    if height[left] < height[right]:
        left += 1
    else:
        right -= 1

  print(total_area)

def compress_matrix(mat):
  row = len(mat)
  col = len(mat[0])
  final = np.zeros((row//2, col//2), dtype=int)

  for m in range(0, row, 2):
    for n in range(0, col, 2):
      total = mat[m][n] + mat[m][n+1] + mat[m+1][n] + mat[m+1][n+1]
      final[m//2][n//2] = total

  return final

def play_game(arena):
  row, col = arena.shape


  mark = 0
  for m in range(row):
    for n in range(col):

      if (arena[m][n] % 50 == 0) and (arena[m][n] >= 50):


        if m > 0 and arena[m-1][n] == 2:
          mark += 2
        if n > 0 and arena[m][n-1] == 2:
          mark += 2
        if m < (row-1) and arena[m+1][n] == 2:
          mark += 2
        if n < (col-1) and arena[m][n+1] == 2:
          mark += 2

  result = f'marks Gained: {mark}.'
  
  if mark >= 10:
    print(f'{result} Your team has survived the game.')
  else:
    print(f'{result} Your team is out.')
