def compress_matrix(mat):


  rows, cols = mat.shape


  comp_rows= rows // 2
  comp_cols= cols // 2


  new_matrix = np.zeros((comp_rows, comp_cols), dtype =int)


  for n in range (0, rows, 2):


        for p in range (0, cols, 2):


            sum = mat[n][p] + mat[n+1][n] + mat[n][p+1] + mat[n+1][p+1]




            new_matrix[n//2][p//2] = sum


  return new_matrix
