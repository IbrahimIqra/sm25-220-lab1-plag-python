def mostWater(arr):
  total_cap = 0
  kick_off = 0
  wrap_up = len(arr)-1
  while kick_off<wrap_up:
    span = wrap_up-kick_off
    height = 0


    if arr[kick_off]<arr[wrap_up]:
      height = arr[kick_off]
    else:
      height = arr[wrap_up]
    capacity = height*span


    if capacity>total_cap:
      total_cap = capacity
    if arr[kick_off]<arr[wrap_up]:
      kick_off+=1
    else:
      wrap_up-=1
  print(total_cap)

def compress_matrix(mat):
  lines, cue = mat.shape
  n_lines = lines//2
  new_cue = cue//2
  n_arr = np.zeros((n_lines,new_cue), dtype=int)
  q = 0
  while q<n_lines:
    p = 0
    while p<new_cue:
      lines_start = q*2
      cue_start = p*2


      sum = (mat[lines_start][cue_start] +
                      mat[lines_start][cue_start+1] +
                      mat[lines_start+1][cue_start] +
                      mat[lines_start+1][cue_start+1])
      n_arr[q][p] = sum
      p+=1
    q+=1
  return n_arr

def play_game(arena):


  lines = len(arena)
  cue = len(arena[0])
  final_score = 0
  q = 0
  while q<lines:
    p = 0
    while p<cue:
      node_value = arena[q][p]
      if node_value>0 and node_value%50== 0:
        point_count = 0


        if q>0 and arena[q-1][p]== 2:
          point_count+=1
        if q < lines-1 and arena[q+1][p]== 2:
          point_count+=1
        if p > 0 and arena[q][p-1]== 2:
          point_count+=1
        if p < cue-1 and arena[q][p+1]== 2:
          point_count+=1
        if point_count>0:
          points_of_node=(node_value//50)+ point_count
          final_score+=points_of_node
      p+=1
    q+=1
  if final_score>=10:
    print(f"Points Gained: {final_score}. Your team has survived the game.")
  else:
