def mostWater(arr):
 


  max = float("-inf")
  for i in range(len(arr)):
    h = arr[i]
    for j in range( i+1, len(arr)):
      w = j-i
      sq = min( h, arr[j]) * w
      if(sq > max):
        max = sq


  print(max)

def compress_matrix(mat):


  row,col=matrix.shape
  r=row//2
  c=col//2


  new_mat= np.zeros((r,c),dtype=matrix.dtype)
  for i in range(0,row,2):
    for j in range(0,col,2):
      sum= matrix[i][j]+matrix[i+1][j]+matrix[i+1][j+1]+matrix[i][j+1]


      new_mat[i//2][j//2]=sum


  return new_mat
