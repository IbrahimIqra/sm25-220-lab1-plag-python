def mostWater(arr):
  #To Do
  leftside=0
  rightside=len(arr)-1
  max_water=0


  while leftside<rightside:
    current_water=min(arr[leftside],arr[rightside])*(rightside-leftside)
    if current_water>max_water:
      max_water=current_water


    elif arr[leftside]< arr[rightside]:
      leftside+=1
    else:
      rightside-=1
  print(max_water)

def play_game(arena):
  row, col = arena.shape
  point = 0
  for i in range(row):
    for j in range(col):
      if arena[i][j] != 0:
        if arena[i][j] % 50 == 0:
          if j-1 >= 0:
            if arena[i][j-1] == 2:
              point += 1
          if i-1 >= 0:
            if arena[i-1][j] ==2:
              point += 1
          if j+1 < col:
            if arena[i][j+1] == 2:
              point += 1
          if i+1 < row:
            if arena[i+1][j] == 2:
              point += 1
  point = point*2
  if point >= 10:
    print(f"Points gained: {point}. Your team has survived the game.")
  else: print(f"Points gained: {point}. Your team is out.")
