def mostWater(arr):
 left=0
 right=len(arr)-1
 max=0


 while left<right:
   if arr[left]<arr[right]:
     wall_height=arr[left]
   else:
     wall_height=arr[right]
   width=right-left
   area=wall_height*width
   if area>max:
     max=area
   if arr[left]<arr[right]:
     left+=1
   else:
     right-=1
 print(max)

def play_game(arena):
 rows=len(arena)
 columns=len(arena[0])
 total=0
 for r in range(rows):
   for c in range(columns):
     if arena[r][c]%50==0 and arena[r][c]!=0:
       points = 0


       if r>0 and arena[r-1][c]==2:
         points+=2
       if c>0 and arena[r][c-1]==2:
         points+=2
       if r<rows-1 and arena[r+1][c]==2:
         points+=2
       if c<columns-1 and arena[r][c+1]==2:
         points+=2
       total+=points
 if total>=10:
     print(f"Points Gained: {total}. Your team has survived the game.")
 else:
     print(f"Points Gained: {total}. Your team is out.")
