def mostWater(arr):
  #To Do
    left, right = 0, len(arr) - 1
    max_area = 0

    while left < right:
        h8 = min(arr[left], arr[right])
        w8 = right - left
        max_area = max(max_area, h8 * w8)

        if arr[left] < arr[right]:
            left+= 1
        else:
            right-=1

    print(max_area)

def compress_matrix(mat):
  #TO DO
  row,coll = mat.shape
  x = int(row/2)
  y = int(coll/2)
  p = 0
  arr = np.zeros((x,y),dtype = int)
  for i in range(0,row,2):
    j = 0
    q = 0
    for j in range(0,coll,2):
      arr[p][q] = mat[i][j]+mat[i+1][j]+mat[i][j+1]+mat[i+1][j+1]
      q+=1
    p+=1

  return arr
