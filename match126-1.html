def mostWater(arr):
  side_l = 0
  side_r = len(arr) - 1
  max_area = 0


  while side_l < side_r:
    if arr[side_l] < arr[side_r]:
      area = arr[side_l]
      side_l += 1
    else:
      area = arr[side_r]
      side_r -= 1
   
    diff = side_r - side_l
    area1 = area * diff


    if area1 > max_area:
      max_area = area1


  print(max_area)

def compress_matrix(mat):
    row, col = mat.shape
    new_mat  = np.zeros((row // 2, col // 2), dtype = int)


    for r in range(0, row, 2):
        for c in range(0, col, 2):
            total = 0


            total += mat[r][c] + mat[r + 1][c] + mat[r][c + 1] + mat[r + 1][c + 1]
            new_mat[r // 2][c // 2] = total


    return new_mat
