def compress_matrix(mat):
  rw, cl = mat.shape
  comp_rw = rw // 2
  comp_cl = cl // 2
  comp_mat = np.zeros((comp_rw, comp_cl), dtype=int)
  for i in range(0, rw, 2):
    for j in range(0, cl, 2):
      blc_sm = (mat[i, j] + mat[i+1, j] + mat[i, j+1] + mat[i+1, j+1])
      comp_mat[i//2, j//2] = blc_sm

  return comp_mat

def play_game(arena):
  rw, cl = arena.shape
  ttl_pnts = 0


  drct = [(-1, 0), (1, 0), (0, -1), (0, 1)]
  for i in range(rw):
    for j in range(cl):
      if arena[i][j] % 50 == 0 and arena[i][j] != 0:


        for direction in drct:
          pi, pj = i + direction[0], j + direction[1]
          if 0 <= pi < rw and 0 <= pj < cl and arena[pi][pj] == 2:
            ttl_pnts += 2

  print(f'Points Gained: {ttl_pnts}.', end=' ')
  if ttl_pnts >= 10:
    print('Your team has survived the game.')
  else:
    print('Your team is out.')
