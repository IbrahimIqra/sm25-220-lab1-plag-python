def row_rotation(exam_week, seat_status):
    total_rotation = exam_week - 1
    total_rows, total_cols = seat_status.shape
    aa_row_index = 0 

    rotation_count = 0
    while rotation_count < total_rotation:
        col_index = 0
        while col_index < total_cols:
            last_item = seat_status[total_rows - 1][col_index]
            row_index = total_rows - 1

            while row_index > 0:
                seat_status[row_index][col_index] = seat_status[row_index - 1][col_index]
                if "AA" in seat_status[row_index][col_index]:
                    aa_row_index = row_index
                row_index = row_index - 1

            seat_status[0][col_index] = last_item
            col_index += 1
        rotation_count += 1

    print(seat_status)
    return aa_row_index + 1

def compress_matrix(matrix):


    total_rows, total_cols = matrix.shape
    compressed_rows, compressed_cols = total_rows // 2, total_cols // 2
    compressed = np.zeros((compressed_rows, compressed_cols), dtype=int)

    i = 0
    row_ptr = 0

    while i < total_rows:
        j = 0
        col_ptr = 0

        while j < total_cols:
            block_sum = matrix[i][j] + matrix[i + 1][j] + matrix[i][j + 1] + matrix[i + 1][j + 1]
            compressed[row_ptr][col_ptr] = block_sum
            col_ptr = col_ptr + 1
            j = j + 2
        row_ptr = row_ptr + 1
        i = i + 2

    return compressed

def play_game(arena):
    total_rows, total_cols = arena.shape
    total_points = 0

    row_index = 0
    while row_index < total_rows:
        col_index = 0
        while col_index < total_cols:
            current_value = arena[row_index][col_index]

          
            if current_value >= 50 and current_value % 50 == 0:

                
                if row_index + 1 < total_rows and arena[row_index + 1][col_index] == 2:
                    total_points += 2

                
                if row_index - 1 >= 0 and arena[row_index - 1][col_index] == 2:
                    total_points += 2

                
                if col_index + 1 < total_cols and arena[row_index][col_index + 1] == 2:
                    total_points += 2

                
                if col_index - 1 >= 0 and arena[row_index][col_index - 1] == 2:
                    total_points += 2

            col_index += 1
        row_index += 1


    if total_points >= 10:
        print(f"Points Gained: {total_points}. Your team has survived the game.")
    else:
        print(f"Points Gained: {total_points}. Your team is out.")
