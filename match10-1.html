 def mostWater(arr):
 
 left = 0
  right = len(arr) - 1
  max_area1 = 0


  while left < right:
    if arr[left] < arr[right]:
      height = arr[left]
    else:
      height = arr[right]
   
    width = right - left
    area = height * width


    if area > max_area1:
      max_area1 = area
    if arr[left] < arr[right]:
      left = left + 1
    else:
      right = right - 1


  print(max_area1)

def play_game(arena):
  #TO DO
    total_points = 0
    rows = len(arena)
    cols = len(arena[0])








    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:
                if i > 0 and arena[i - 1][j] == 2:           # Up
                    total_points += 1
                if i < rows - 1 and arena[i + 1][j] == 2:    # Down
                    total_points += 1
                if j > 0 and arena[i][j - 1] == 2:           # Left
                    total_points += 1
                if j < cols - 1 and arena[i][j + 1] == 2:    # Right
                    total_points += 1




    print('Alice in Borderland:')
    final_points = total_points * 2
    if final_points >= 10:
        print(f"Points Gained: {final_points}. Your team has survived the game.")
    else:
        print(f"Points Gained: {final_points}. Your team is out.")
