def compress_matrix(mat):
  r = len(mat)
  c = len(mat[0])


  new_r = r // 2
  new_c = c // 2
  a = np.zeros((new_r, new_c), dtype=int)
  for i in range(0, r, 2):
      for j in range(0, c, 2):
          sum = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]


          a[i//2][j//2] = sum
  return a
