def mostWater(arr):
    max_area = 0
    left = 0
    right = len(arr) - 1
    
    while left < right:


        current_height = min(arr[left], arr[right])
        current_width = right - left
        current_area = current_height * current_width
        if current_area > max_area:
            max_area = current_area
        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1
    
    print(max_area)

def row_rotation(exam_week, seat_status):
    for _ in range(exam_week):
        last_row = seat_status[-1].copy()
        for i in range(len(seat_status) - 1, 0, -1):
            seat_status[i] = seat_status[i - 1]
        seat_status[0] = last_row
    
    print_matrix(seat_status)
    
    for i in range(len(seat_status)):
        if 'AA' in seat_status[i]:
            return i + 1  
    return None

def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0]) if rows > 0 else 0
    compressed_rows = rows // 2
    compressed_cols = cols // 2
    compressed = np.array([[0] * compressed_cols for _ in range(compressed_rows)])
    


    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            total = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            compressed[i//2][j//2] = total
    
    return compressed

def play_game(arena):
    total_points = 0
    rows = len(arena)


    cols = len(arena[0]) if rows > 0 else 0
    
    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 360 == 0 and arena[i][j] != 0:
                # Check up
                if i > 0 and arena[i-1][j] == 2:
                    total_points += 2
                # Check down
                if i < rows - 1 and arena[i+1][j] == 2:
                    total_points += 2
                # Check left
                if j > 0 and arena[i][j-1] == 2:
                    total_points += 2
                # Check right
                if j < cols - 1 and arena[i][j+1] == 2:
                    total_points += 2
