def row_rotation(exam_week, seat_layout):
    rotations_needed = exam_week - 1
    num_rows, num_cols = seat_layout.shape
    aa_position = 0

    current_rotation = 0
    while current_rotation < rotations_needed:
        col = 0
        while col < num_cols:
            bottom_value = seat_layout[num_rows - 1][col]
            row = num_rows - 1

            while row > 0:
                seat_layout[row][col] = seat_layout[row - 1][col]
                               if seat_layout[row][col] == "AA":
                    aa_position = row
                row -= 1

            seat_layout[0][col] = bottom_value
            col += 1

        current_rotation += 1

    print(seat_layout)


    return aa_position + 1  # 1-based index

def compress_matrix(input_matrix):
    rows, cols = input_matrix.shape
    half_rows, half_cols = rows // 2, cols // 2
    result_matrix = np.zeros((half_rows, half_cols), dtype=int)

    row_index = 0
    new_r = 0

    while row_index < rows:
        col_index = 0
        new_c = 0

        while col_index < cols:
            
            current_sum = (
                input_matrix[row_index][col_index] +
                input_matrix[row_index + 1][col_index] +
                input_matrix[row_index][col_index + 1] +
                input_matrix[row_index + 1][col_index + 1]
            )

            result_matrix[new_r][new_c] = current_sum
            new_c += 1
            col_index += 2

        new_r += 1
        row_index += 2

    return result_matrix

def play_game(game_board):
    rows, cols = game_board.shape
    points = 0

    r = 0
    while r < rows:
        c = 0
        while c < cols:
            cell = game_board[r][c]

                       if not cell % 50 and cell >= 50:

                
                above = (r > 0 and game_board[r - 1][c] == 2)
                below = (r < rows - 1 and game_board[r + 1][c] == 2)
                left = (c > 0 and game_board[r][c - 1] == 2)
                right = (c < cols - 1 and game_board[r][c + 1] == 2)

                points += 2 * (above + below + left + right)

            c += 1
        r += 1

   
    status = "survived the game" if points >= 10 else "is out"
    print(f"Points Gained: {points}. Your team has {status}.")
