def mostWater(arr):
  
    max_area, l, r = 0, 0, len(arr) - 1 



    while l < r:
        if arr[l] < arr[r]:
            height = arr[l]
        else:
            height = arr[r]

        width = r - l
        area = height * width
        if area > max_area:
            max_area = area

        if arr[l] < arr[r]:
            l += 1
        else:
            r -= 1

    print(max_area)
