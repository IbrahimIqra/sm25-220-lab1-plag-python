def mostWater(arr):
  left =  0
  right= len(arr) - 1
  count =0

  while left < right:
    diff = right - left
    if arr[left] < arr[right]:
      area = arr[left] * diff
      if count < area:
        count = area
    elif arr[left]> arr[right]:
      area = arr[right] *diff
      if count< area:


        count =  area

    if arr[left] < arr[right]:
      left += 1
    else:
      right -= 1

  print(count)

def compress_matrix(matrix):
  rows = len(matrix)
  cols = len(matrix[0])



  new_row = rows // 2
  new_col= cols // 2
  new_matrix=np.zeros((new_row,new_col),dtype=int)

  for i in range(0, rows, 2):
    for j in range(0, cols, 2):


      new_matrix[i // 2][j // 2] = (matrix[i][j] +matrix[i+1][j] +matrix[i][j+1] + matrix[i + 1][j + 1])

  return new_matrix
