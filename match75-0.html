def mostWater(arr):
    lf= 0
    rt= len(arr) - 1
    max_area= 0

    while lf < rt:


        if arr[lf] < arr[rt]:
            height = arr[lf]
        else:
            height= arr[rt]
        width= rt - lf
        area= height * width
        if area > max_area:
            max_area = area
        if arr[lf] < arr[rt]:
            lf+= 1
        else:
            rt-=1
    print(max_area)

def compress_matrix(mat):
    r= len(mat)
    c= len(mat[0])
    new_rows = r// 2
    new_cols = c// 2
    result = np.zeros((new_rows, new_cols), dtype=int)   
    for i in range(0,r,2):


        for j in range(0,c,2):
            sum = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            result[i//2][j//2]=sum
    return np.array(result)

def play_game(arena):
    row=len(arena)
    col=len(arena[0])


    total = 0
    for i in range(row):
       for j in range(col):
          if arena[i][j] % 50 ==0 and arena[i][j]!= 0:
            p=0
            if i>0 and arena[i-1][j] == 2:
                    p += 1
            if i<row - 1 and arena[i+1][j] == 2:
                    p+= 1
            if j>0 and arena[i][j-1] == 2:
                    p+= 1
            if j<col - 1 and arena[i][j+1]== 2:
                    p+= 1

            total+= p

    total*= 2
    if total>= 10:
        print("Points Gained:", total, ". Your team has survived the game.")
    else:
        print("Points Gained:",total, ". Your team is out.")
