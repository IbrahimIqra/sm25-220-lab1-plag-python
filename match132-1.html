def mostWater(arr):
   left = 0
   right = len(arr) - 1
   max_area = 0


   while left < right:
       # Calculate the area
       height = min(arr[left], arr[right])
       width = right - left
       area = height * width


       # Update max_area if the current area is larger
       max_area = max(max_area, area)


       # Move the pointer from the shorter line
       if arr[left] < arr[right]:
           left += 1
       else:
           right -= 1


   print(max_area)

def play_game(arena):
 row,column=arena.shape
 sum=0
 for i in range(row):
   for j in range(column):
     if arena[i][j]!=0 and arena[i][j]%50==0:
       if i-1>=0 and arena[i-1][j]==2:
         sum+=1
       if j+1<=column-1 and arena[i][j+1]==2:
         sum+=1
       if i+1<=row-1 and arena[i+1][j]==2:
         sum+=1
       if j-1>=0 and arena[i][j-1]==2:
         sum+=1
 if sum*2<10:
    print(f'Points Gained: {sum*2}. Your team has survived the game.')
 else:
