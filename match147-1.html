import numpy as np
def mostWater(arr):
  max=0
  x=len(arr)
  for i in range(0,x):
    for j in range(i+1,x):


      if arr[i] < arr[j]:
        h=arr[i]
      else :
        h= arr[j]
      w = j-i 
      area= h*w
      if area>max:
        max=area 
  print(max)

import numpy as np
def compress_matrix(mat):
  row=len(mat)
  coloumn=len(mat[0])
  compressed_row= row//2
  compressed_coloumn= coloumn//2
  new_matrix= np.zeros((compressed_row,compressed_coloumn), dtype=int)
  for i in range(0,row,2):
    for j in range(0,coloumn,2):
      sum= mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
      new_matrix[i//2][j//2]=sum
  return new_matrix

import numpy as np


def play_game(arena):
  row= len(arena)
  coloumn= len(arena[0])
  point=0
  for i in range(0,row):
    for j in range(0, coloumn):
      cell=arena[i][j]
      if cell%50==0 and cell!=0 : #cause 0 is also a multiple of 50,so I added it.
         


         if i-1>=0 and arena[i-1][j]==2 : #Up (keepin track)
          point+=2
         if i+1<row and arena[i+1][j]==2 : #Down
          point+=2


         if j-1>=0 and arena[i][j-1]==2 : #Left
          point+= 2 
         if j+1<coloumn and arena[i][j+1]==2 : #Right
          point+=2
  if point>=10 :
    print(f"Points Gained: {point}. Your team has survived the game.")
  else :
    print(f"Points Gained: {point}. Your team is out.")
