import numpy as np
def mostWater(arr):
  maxCap=0

  p1=0
  p2=len(arr)-1

  while p1<p2:


    if arr[p1]<arr[p2]:
      smaller_h=arr[p1]
    else:
      smaller_h=arr[p2]
    w=p2-p1
    possible_cap=smaller_h*w

    if possible_cap > maxCap:
     maxCap = possible_cap

    if arr[p1] < arr[p2]:
      p1 += 1
    else:
      p2 -= 1
  #return np.array(maxCap)
  print(maxCap)
