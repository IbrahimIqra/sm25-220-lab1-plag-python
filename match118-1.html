#Task 1
def max_area(height):
    max_water = 0
    left = 0
    right = len(height) - 1
    while left < right:
        current_water = min(height[left], height[right]) * (right - left)


        if current_water > max_water:
            max_water = current_water
        if height[left] < height[right]:
            left += 1
        else:
            right -= 1
    return max_water

def row_rotation(exam_week, seat_status):
    rows = len(seat_status)
    rotation = exam_week % rows


    for _ in range(rotation):
        last_row = seat_status[-1]
        for i in range(rows-1, 0, -1):
            seat_status[i] = seat_status[i-1]
        seat_status[0] = last_row
    
    for i in range(rows):
        if "AA" in seat_status[i]:
            return i
    return -1

def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0])
    compressed = [[0]*(cols//2) for _ in range(rows//2)]


    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            total = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1]
            compressed[i//2][j//2] = total
    return compressed

def game_survival(matrix):
    total_points = 0
    rows = len(matrix)


    cols = len(matrix[0]) if rows > 0 else 0
    
    for i in range(rows):
        for j in range(cols):
            if matrix[i][j] % 50 == 0 and matrix[i][j] != 0:
                points = 0
                if i > 0 and matrix[i-1][j] == 2:
                    points += 1
                if i < rows-1 and matrix[i+1][j] == 2:
                    points += 1


                if j > 0 and matrix[i][j-1] == 2:
                    points += 1
                if j < cols-1 and matrix[i][j+1] == 2:
                    points += 1
                total_points += points * 2
    
    if total_points >= 10:
        return f"Points Gained: {total_points}. Your team has survived the game."
    else:
        return f"Points Gained: {total_points}. Your team is out."
