def mostWater(arr):
    left = 0
    right = len(arr) - 1
    max_areaa = 0


    while left < right:
        height = min(arr[left], arr[right])
        width = right - left
        area = height * width
        max_areaa = max(max_areaa, area)


        if arr[left] < arr[right]:
            left += 1
        else:
            right -= 1


    print(max_areaa)

import numpy as np


def print_matrix(matrix):
    for row in matrix:
        print("| ", end="")
        for val in row:
            print(f"{val:3}", end=" | ")
        print("\n" + "-" * (6 * len(row)))




def compress_matrix(mat):
    rows = len(mat)
    cols = len(mat[0])
    compressed = np.zeros((rows // 2, cols // 2), dtype=int)


    for i in range(0, rows, 2):
        for j in range(0, cols, 2):
            block_sum = (
                mat[i][j] + mat[i][j + 1] +
                mat[i + 1][j] + mat[i + 1][j + 1]
            )
            compressed[i // 2][j // 2] = block_sum


    return compressed

import numpy as np


def play_game(arena):
    rows, cols = arena.shape
    points = 0


   


    for i in range(rows):
        for j in range(cols):
            if arena[i][j] % 50 == 0 and arena[i][j] != 0:
               
                for dx, dy in [(-1,0), (1,0), (0,-1), (0,1)]:


                    ni, nj = i + dx, j + dy
                    if 0 <= ni < rows and 0 <= nj < cols and arena[ni][nj] == 2:
                        points += 2


    result = "Your team has survived the game." if points >= 10 else "Your team is out."
    print(f"Points Gained: {points}. {result}")
