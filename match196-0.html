#Task01 : Container with Most Water



def mostWater(arr):
  i = 0
  j = len(arr) - 1
  max_area = 0

  while i < j:
    if arr[i] < arr[j] :
      area = arr[i]*(j-i)
    else:
      area = arr[j]*(j-i)



    if area > max_area:
      max_area = area

    if arr[i] < arr[j]:
      i += 1
    else:
      j -= 1
  print(max_area)


#Task02 : Row Rotation Policy of Your Classroom

def row_rotation(exam_week, seat_status):
  row, col = seat_status.shape
  row_shift = (exam_week - 1) % row 
  rotated_seat_status = np.array(seat_status)
  for i in range(row):
    rotated_seat_status[(i+row_shift)%row] = seat_status[i]
  
  AA_pos = -1
  for i in range(row):
    for j in range(col):
      if rotated_seat_status[i][j] == 'AA':
        AA_pos = i+1
        break
  return AA_pos


#Task 04: Matrix Compression

def compress_matrix(mat):
  row,col = mat.shape


  c_row,c_col = row//2,col//2
  c_mat = np.zeros((c_row,c_col),dtype=int)
  for i in range(0,row,2):
    for j in range(0,col,2):
      c_mat[i//2][j//2] = mat[i][j] + mat[i+1][j] + mat[i][j+1] + mat[i+1][j+1]
      
  return c_mat


#Task 05: Game Arena

def play_game(arena):
  row,col = arena.shape


  points = 0

  for i in range(row):
    for j in range(col):
      if arena[i][j] % 50 == 0 and arena[i][j] != 0:


        if i > 0 and arena[i-1][j] == 2:
          points += 2
        if i < row -1 and arena[i+1][j] == 2:
          points += 2
        if j > 0 and arena[i][j-1] == 2:
          points += 2
        if j < col - 1 and arena[i][j+1] == 2:
          points += 2

  if points >= 10:
    print(f'Points Gained: {points}. Your team has survived the game.')
  else:
    print(f'Points Gained: {points}. Your team is out')
