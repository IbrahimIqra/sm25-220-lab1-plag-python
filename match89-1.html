def mostWater(arr):
  left = 0
  right = len(arr) - 1
  mostWater = 0
  while left < right:


    if arr[left] < arr[right]:
      height = arr[left]
    else:
      height = arr[right]
    width = right - left
    currentWater = height*width


    if currentWater > mostWater:
      mostWater = currentWater
    else:
      pass


    if arr[left] < arr[right]:
      left+=1
    else:
      right-=1
  print(mostWater)

def row_rotation(exam_week, seat_status):
  row_c= len(seat_status)
  column_c=len(seat_status[0])


  new=np.array([[None]*column_c for _ in range(row_c)])




  for old_row in range(row_c):
    new_row=(old_row + exam_week) % row_c


  for column in range(column_c):
    new[new_row][column]=seat_status[old_row][column]


  for i in range(row_c):
    for j in range(column_c):
      seat_status[i][j]=new[i][j]


  for i in range(row_c):
    for j in range(column_c):
      if seat_status[i][j]== "AA":
         return i

def play_game(arena):
  r_count=len(arena)
  c_count=len(arena[0])


  points=0




  for i in range(r_count):
    for j in range(c_count):
      if arena[i][j]%50==0 and arena[i][j]!=0:


        if i>0 and arena[i-1][j]==2:
          points+=2
        if i<r_count-1 and arena[i+1][j]==2:
          points+=2
        if j>0 and arena[i][j-1]==2:
          points+=2
        if j<c_count-1 and arena[i][j+1]==2:
          points+=2
