Assignment Task 1: Container with Most Water


def mostWater(arr):
  a=len(arr)
  max_area=0


  for i in range(a):
    for j in range(i+1,a):


      if arr[i]<arr[j]:
        h_eight=arr[i]
      else:
        h_eight=arr[j]


      width=j-i   #width between the two holes


      area= h_eight*width


      if area > max_area:
        max_area= area


  print(max_area)

def compress_matrix(mat):
 r=len(mat) #row count
 c=len(mat[0]) #column count
 new_row,new_col=r//2, c//2
 result_arr=np.zeros((new_row,new_col),dtype=int)


 for i in range(r):
  for j in range(c):


    if i%2==0 and j%2==0:
      sum=mat[i][j]+mat[i+1][j]+mat[i][j+1]+mat[i+1][j+1]

 return result_arr

def play_game(arena):
  r=len(arena) #row count
  c=len(arena[0]) #column count


  score=0


  for i in range(r):
    for j in range(c):
      if arena[i][j]%50==0 and arena[i][j]!=0:
        if i>0 and arena[i-1][j]==2:
          score+=arena[i-1][j]
        if i<r-1 and arena[i+1][j]==2:
          score+=arena[i+1][j]


        if j>0 and arena[i][j-1]==2:
          score+=2
        if j<c-1 and arena[i][j+1]==2:
          score+=2


  if score<10:
    print(f'Points Gained: {score}. Your team is out.')
