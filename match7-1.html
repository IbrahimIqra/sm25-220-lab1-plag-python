def mostWater(arr):
    maximum_area = 0
    right_line, left_line =  len(arr) - 1 ,0


    while left_line < right_line:
        if arr[left_line] < arr[right_line]:
          h = arr[left_line]
        else:
          h = arr[right_line] 
        
        w = right_line - left_line
        area = h * w
        if  maximum_area < area:
            maximum_area = area
        if  arr[right_line] >arr[left_line] :
            left_line += 1
        else:
            right_line -= 1
    print(maximum_area)
