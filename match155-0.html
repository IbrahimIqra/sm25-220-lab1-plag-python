def mostWater(arr):
    maxarea=0
    for i in range(len(arr)):
        for j in range(i+1,len(arr)):
            h=min(arr[i],arr[j])
            w=j-i
            area=h*w
            maxarea=max(maxarea,area)
    print(maxarea)

def row_rotation(exam_week, seat_status):
    row,col=seat_status.shape
    rot=(exam_week-1)%row


    new_mat=np.empty((row,col),dtype=object)


    for i in range(row):
        new_row=(i+rot)%row
        for j in range(col):
            new_mat[new_row][j]=seat_status[i][j]




    for i in range(row):
        for j in range(col):
            if new_mat[i][j]=='AA':
                return i+1

def compress_matrix(mat):
    row,col=mat.shape


    newrow=row//2
    newcol=col//2


    comMat=np.empty((newrow,newcol),dtype=object)


    for i in range(0,row,2):
        for j in range(0,col,2):
            val=mat[i][j]+mat[i][j+1]+mat[i+1][j]+mat[i+1][j+1]
            comMat[i//2][j//2]=val


    return comMat

def play_game(arena):


    row,col=arena.shape
    points=0
    nearCells=[(-1,0),(1,0),(0,-1),(0,1)]


    for i in range(row):
        for j in range(col):
            if arena[i][j]%50==0 and arena[i][j]!=0:
                for x,y in nearCells:


                    newi,newj = i+x,j+y
                    if 0<=newi<row and 0<=newj<col:
                        if arena[newi][newj]==2:
                            points+=2


    if points>=10:
        print(f"Points Gained: {points}. Your team has survived the game.")
    else:
        print(f"Points Gained: {points}. Your team is out.")
