import numpy as np
def mostWater(arr):
  maximum=0
  l_idx=0
  r_idx=len(arr)-1
  #distance=(r_idx - l_idx)
  while l_idx< r_idx:
    minimum= min(arr[l_idx],arr[r_idx])*(r_idx - l_idx)
    maximum=max(maximum,minimum)
    if arr[l_idx]<arr[r_idx]:
       l_idx+=1
    else:
      r_idx-=1
  return maximum

   row=0
