Task 01
def mostWater(arr):
    n = len(arr)
    res = 0
    for i in range(n):
        for j in range(i + 1, n):              
            amount = min(arr[i], arr[j]) * (j - i)                     
            res = max(amount, res)
    return res

def compress_matrix(mat):
    row,col=mat.shape
    arr=np.zeros((2,2),dtype=int)
    r=row//2
    c=col//2
    for i in range(r):
        for j in range(c):
            arr[0][0]+=mat[i][j]
    for i in range(r):
        for j in range(c,col):
            arr[0][1]+=mat[i][j]
    for i in range(r,row):
        for j in range(c):
            arr[1][0]+=mat[i][j]
    for i in range(r,row):
        for j in range(c,col):
            arr[1][1]+=mat[i][j]
    return arr
