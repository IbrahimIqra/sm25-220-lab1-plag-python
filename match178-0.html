#Container with most water
def mostWater(arr):
    maxarea = 0
    left_side = 0
    right_side = len(arr) - 1 #8

    while left_side < right_side: # 0 < 8
        height = min(arr[left_side], arr[right_side])
        width = right_side - left_side
        area = height * width
        if area > maxarea:
            maxarea = area
        if arr[left_side] < arr[right_side]:
            left_side = left_side +1
        else:
            right_side = right_side -1
    print(maxarea)

def row_rotation(exam_week, seat_status):
    rows = len(seat_status)
    cols = len(seat_status[0])
    actual_week = (exam_week - 1) % rows  

    rotated_mat = [["" for m in range(cols)] for m in range(rows)] #or rotated = [[""] * cols for m in range(rows)]



    for i in range(rows):
        new = (i + actual_week) % rows
        for j in range(cols):
            rotated_mat[new][j] = seat_status[i][j]

    for i in rotated_mat:
        print("|", " | ".join(i), "|")

    for i in range(rows):
        for j in range(cols):
            if rotated_mat[i][j] == "AA":
                print(f"\nYour friend AA will be on row {i + 1}")
                return

def play_game(arena):
    row = len(arena) #4
    col= len(arena[0]) #4


    totalpoints = 0

    for mp in range(row): #3
        for sp in range(col): #1
            if arena[mp][sp] % 50 == 0 and arena[mp][sp] != 0:


                if mp > 0 and arena[mp-1][sp] == 2:
                    totalpoints += 1
                if mp < row - 1 and arena[mp+1][sp] == 2:
                    totalpoints += 1
                if sp > 0 and arena[mp][sp-1] == 2:
                    totalpoints += 1
                if sp < col - 1 and arena[mp][sp+1] == 2:
                    totalpoints += 1

    if totalpoints*2 < 10:
        print(f"Points Gained: {totalpoints*2}. Your team is out.")
    else:
        print(f"Points Gained: {totalpoints*2}. Your team has survived the game.")
